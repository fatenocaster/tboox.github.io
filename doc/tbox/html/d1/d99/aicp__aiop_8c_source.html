209 static tb_bool_t tb_aiop_push_acpt(tb_aiop_ptor_impl_t* impl, tb_aice_t const* aice)
210 {
211  // check
212  tb_assert_and_check_return_val(impl && aice, tb_false);
213  tb_assert_and_check_return_val(aice->code == TB_AICE_CODE_ACPT, tb_false);
214 
215  // the aico
216  tb_aiop_aico_t* aico = (tb_aiop_aico_t*)aice->aico;
217  tb_assert_and_check_return_val(aico && aico->base.handle, tb_false);
218 
219  // the priority
220  tb_size_t priority = tb_aice_impl_priority(aice);
221  tb_assert_and_check_return_val(priority < tb_arrayn(impl->spak) && impl->spak[priority], tb_false);
222 
223  // init the acpt aice
224  tb_aice_t acpt_aice = *aice;
225  acpt_aice.state = TB_STATE_OK;
226 
227  // done
228  tb_size_t list_indx = 0;
229  tb_size_t list_size = 0;
230  tb_socket_ref_t list_sock[2048];
231  tb_ipv4_t list_addr[2048];
232  tb_uint16_t list_port[2048];
233  tb_size_t list_maxn = tb_arrayn(list_sock);
234  tb_socket_ref_t acpt = (tb_socket_ref_t)aico->base.handle;
235  tb_queue_ref_t spak = impl->spak[priority];
236  tb_socket_ref_t sock = tb_null;
237  do
238  {
239  // accept it
240  for (list_size = 0; list_size < list_maxn && (list_sock[list_size] = tb_socket_accept(acpt, list_addr + list_size, list_port + list_size)); list_size++) ;
241 
242  // enter
243  tb_spinlock_enter(&impl->lock);
244 
245  // push some acpt aice
246  for (list_indx = 0; list_indx < list_size && (sock = list_sock[list_indx]); list_indx++)
247  {
248  // init aico
249  acpt_aice.u.acpt.aico = tb_aico_init(aico->base.aicp);
250 
251  // trace
252  tb_trace_d("push: acpt[%p]: sock: %p, aico: %p", aico->base.handle, sock, acpt_aice.u.acpt.aico);
253 
254  // open aico and push the acpt aice if not full?
255  if ( acpt_aice.u.acpt.aico
256  && tb_aico_open_sock(acpt_aice.u.acpt.aico, sock)
257  && !tb_queue_full(spak))
258  {
259  // save addr
260  acpt_aice.u.acpt.addr = list_addr[list_indx];
261 
262  // save port
263  acpt_aice.u.acpt.port = list_port[list_indx];
264 
265  // push to the spak queue
266  tb_queue_put(spak, &acpt_aice);
267  }
268  else
269  {
270  // close the left sock
271  tb_size_t i;
272  for (i = list_indx; i < list_size; i++)
273  {
274  // close it
275  if (list_sock[i]) tb_socket_exit(list_sock[i]);
276  list_sock[i] = tb_null;
277  }
278 
279  // exit aico
280  if (acpt_aice.u.acpt.aico) tb_aico_exit(acpt_aice.u.acpt.aico);
281  acpt_aice.u.acpt.aico = tb_null;
282 
283  // trace
284  tb_trace_e("push: acpt failed!");
285  break;
286  }
287  }
288 
289  // leave
290  tb_spinlock_leave(&impl->lock);
291 
292  } while (list_indx == list_maxn);
293 
294  // ok
295  return tb_true;

297 static tb_pointer_t tb_aiop_spak_loop(tb_cpointer_t priv)
298 {
299  // check
300  tb_aiop_ptor_impl_t* impl = (tb_aiop_ptor_impl_t*)priv;
301  tb_aicp_impl_t* aicp = impl? impl->base.aicp : tb_null;
302 
303  // done
304  do
305  {
306  // check
307  tb_assert_and_check_break(impl && impl->aiop && impl->list && impl->timer && impl->ltimer && aicp);
308 
309  // trace
310  tb_trace_d("loop: init");
311 
312  // loop
313  while (!tb_atomic_get(&aicp->kill))
314  {
315  // the delay
316  tb_size_t delay = tb_timer_delay(impl->timer);
317 
318  // the ldelay
319  tb_size_t ldelay = tb_ltimer_delay(impl->ltimer);
320  tb_assert_and_check_break(ldelay != -1);
321 
322  // trace
323  tb_trace_d("loop: wait: ..");
324 
325  // wait aioe
326  tb_long_t real = tb_aiop_wait(impl->aiop, impl->list, impl->maxn, tb_min(delay, ldelay));
327 
328  // trace
329  tb_trace_d("loop: wait: %ld", real);
330 
331  // spak ctime
332  tb_cache_time_spak();
333 
334  // spak timer
335  if (!tb_timer_spak(impl->timer)) break;
336 
337  // spak ltimer
338  if (!tb_ltimer_spak(impl->ltimer)) break;
339 
340  // killed?
341  tb_check_break(real >= 0);
342 
343  // error? out of range
344  tb_assert_and_check_break(real <= impl->maxn);
345 
346  // timeout?
347  tb_check_continue(real);
348 
349  // grow it if aioe is full
350  if (real == impl->maxn)
351  {
352  // grow size
353  impl->maxn += (aicp->maxn >> 4) + 16;
354  if (impl->maxn > aicp->maxn) impl->maxn = aicp->maxn;
355 
356  // grow list
357  impl->list = tb_ralloc(impl->list, impl->maxn * sizeof(tb_aioe_t));
358  tb_assert_and_check_break(impl->list);
359  }
360 
361  // walk aioe list
362  tb_size_t i = 0;
363  tb_bool_t end = tb_false;
364  for (i = 0; i < real && !end; i++)
365  {
366  // the aioe
367  tb_aioe_t const* aioe = &impl->list[i];
368  tb_assert_and_check_break_state(aioe, end, tb_true);
369 
370  // the aice
371  tb_aice_t const* aice = aioe->priv;
372  tb_assert_and_check_break_state(aice, end, tb_true);
373 
374  // the aico
375  tb_aiop_aico_t* aico = (tb_aiop_aico_t*)aice->aico;
376  tb_assert_and_check_break_state(aico, end, tb_true);
377 
378  // have wait?
379  tb_check_continue(aice->code);
380 
381  // have been waited ok for the timer timeout/killed func? need not spak it repeatly
382  tb_check_continue(!aico->wait_ok);
383 
384  // sock?
385  if (aico->base.type == TB_AICO_TYPE_SOCK)
386  {
387  // push the acpt aice
388  if (aice->code == TB_AICE_CODE_ACPT) end = tb_aiop_push_acpt(impl, aice)? tb_false : tb_true;
389  // push the sock aice
390  else end = tb_aiop_push_sock(impl, aice)? tb_false : tb_true;
391  }
392  else if (aico->base.type == TB_AICO_TYPE_FILE)
393  {
394  // poll file
395  tb_aicp_file_poll(impl);
396  }
397  else tb_assert(0);
398  }
399 
400  // end?
401  tb_check_break(!end);
402 
403  // work it
404  tb_aiop_spak_work(impl);
405  }
406 
407  } while (0);
408 
409  // trace
410  tb_trace_d("loop: exit");
411 
412  // kill
413  tb_aicp_kill((tb_aicp_ref_t)aicp);
414 
415  // exit
416  tb_thread_return(tb_null);
417  return tb_null;
418 }

528  aico->bltimer = 1;
529  }
530 
531  // ok
532  ok = tb_true;
533 
534  } while (0);
535 
536  // failed? restore it
537  if (!ok)
538  {
539  // trace
540  tb_trace_d("wait: aico: %p, code: %lu: failed", aico, aice->code);
541 
542  // restore it
543  aico->aice = prev;
544  aico->waiting = 0;
545  }
546 
547  // ok?
548  return ok;
