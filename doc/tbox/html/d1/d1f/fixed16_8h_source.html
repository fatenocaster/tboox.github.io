381 static __tb_inline__ tb_fixed16_t tb_fixed16_mul_int32(tb_fixed16_t x, tb_fixed16_t y)
382 {
383  // get sign
384  tb_int32_t s = tb_int32_get_sign(x ^ y);
385  x = tb_fixed16_abs(x);
386  y = tb_fixed16_abs(y);
387 
388  tb_uint32_t xh = x >> 16;
389  tb_uint32_t xl = x & 0xffff;
390  tb_uint32_t yh = y >> 16;
391  tb_uint32_t yl = y & 0xffff;
392 
393  tb_uint32_t r = xh * y + xl * yh + (xl * yl >> 16);
394  return tb_int32_set_sign(r, s);
395 }
