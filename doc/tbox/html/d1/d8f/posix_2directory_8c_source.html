101 static tb_void_t tb_directory_walk_impl(tb_char_t const* path, tb_bool_t recursion, tb_bool_t prefix, tb_directory_walk_func_t func, tb_cpointer_t priv)
102 {
103  // check
104  tb_assert_and_check_return(path && func);
105 
106  // last
107  tb_long_t last = tb_strlen(path) - 1;
108  tb_assert_and_check_return(last >= 0);
109 
110  // init info
111  tb_char_t temp[4096] = {0};
112  DIR* directory = tb_null;
113  if ((directory = opendir(path)))
114  {
115  // walk
116  struct dirent* item = tb_null;
117  while ((item = readdir(directory)))
118  {
119  // check
120  tb_assert_and_check_continue(item->d_name && item->d_reclen);
121 
122  // the item name
123  tb_char_t name[1024] = {0};
124  tb_strlcpy(name, item->d_name, tb_min(item->d_reclen, 1023));
125  if (tb_strcmp(name, ".") && tb_strcmp(name, ".."))
126  {
127  // the temp path
128  tb_long_t n = tb_snprintf(temp, 4095, "%s%s%s", path, path[last] == '/'? "" : "/", name);
129  if (n >= 0) temp[n] = '\0';
130 
131  // the file info
132  tb_file_info_t info = {0};
133  if (tb_file_info(temp, &info))
134  {
135  // do callback
136  if (prefix) func(temp, &info, priv);
137 
138  // walk to the next directory
139  if (info.type == TB_FILE_TYPE_DIRECTORY && recursion) tb_directory_walk_impl(temp, recursion, prefix, func, priv);
140 
141  // do callback
142  if (!prefix) func(temp, &info, priv);
143  }
144  }
145  }
146 
147  // exit directory
148  closedir(directory);
149  }

155 {
156  // check
157  tb_assert_and_check_return_val(path, tb_false);
158 
159  // the full path
160  tb_char_t full[TB_PATH_MAXN];
161  path = tb_path_full(path, full, TB_PATH_MAXN);
162  tb_assert_and_check_return_val(path, tb_false);
163 
164  // make it
165  tb_bool_t ok = !mkdir(path, S_IRWXU)? tb_true : tb_false;
166  if (!ok)
167  {
168  // make directory
169  tb_char_t temp[TB_PATH_MAXN] = {0};
170  tb_char_t const* p = full;
171  tb_char_t* t = temp;
172  tb_char_t const* e = temp + TB_PATH_MAXN - 1;
173  for (; t < e && *p; t++)
174  {
175  *t = *p;
176  if (*p == '/')
177  {
178  // make directory if not exists
179  if (!tb_file_info(temp, tb_null)) mkdir(temp, S_IRWXU);
180 
181  // skip repeat '/'
182  while (*p && *p == '/') p++;
183  }
184  else p++;
185  }
186 
187  // make it again
188  ok = !mkdir(path, S_IRWXU)? tb_true : tb_false;
189  }
190 
191  // ok?
192  return ok;
193 }
