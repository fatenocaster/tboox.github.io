48 tb_handle_t tb_file_init(tb_char_t const* path, tb_size_t mode)
49 {
50  // check
51  tb_assert_and_check_return_val(path, tb_null);
52 
53  // the full path
54  tb_char_t full[TB_PATH_MAXN];
55  path = tb_path_full(path, full, TB_PATH_MAXN);
56  tb_assert_and_check_return_val(path, tb_null);
57 
58  // flags
59  tb_size_t flags = 0;
60  if (mode & TB_FILE_MODE_RO) flags |= O_RDONLY;
61  else if (mode & TB_FILE_MODE_WO) flags |= O_WRONLY;
62  else if (mode & TB_FILE_MODE_RW) flags |= O_RDWR;
63 
64  if (mode & TB_FILE_MODE_CREAT) flags |= O_CREAT;
65  if (mode & TB_FILE_MODE_APPEND) flags |= O_APPEND;
66  if (mode & TB_FILE_MODE_TRUNC) flags |= O_TRUNC;
67 
68  // dma mode, no cache
69 #ifdef TB_CONFIG_OS_LINUX
70  if (mode & TB_FILE_MODE_DIRECT) flags |= O_DIRECT;
71 #endif
72 
73  // for native aio aicp
74 #if defined(TB_CONFIG_ASIO_HAVE_NAIO)
75  if (mode & TB_FILE_MODE_AICP) flags |= O_DIRECT;
76 #endif
77 
78  // noblock
79  flags |= O_NONBLOCK;
80 
81  // modes
82  tb_size_t modes = 0;
83  if (mode & TB_FILE_MODE_CREAT)
84  {
85  //if ((mode & TB_FILE_MODE_RO) | (mode & TB_FILE_MODE_RW)) modes |= S_IREAD;
86  //if ((mode & TB_FILE_MODE_WO) | (mode & TB_FILE_MODE_RW)) modes |= S_IWRITE;
87  modes = 0777;
88  }
89 
90  // open it
91  tb_long_t fd = open(path, flags, modes);
92  if (fd < 0 && (mode & TB_FILE_MODE_CREAT))
93  {
94  // make directory
95  tb_char_t temp[TB_PATH_MAXN] = {0};
96  tb_char_t const* p = full;
97  tb_char_t* t = temp;
98  tb_char_t const* e = temp + TB_PATH_MAXN - 1;
99  for (; t < e && *p; t++)
100  {
101  *t = *p;
102  if (*p == '/')
103  {
104  // make directory if not exists
105  if (!tb_file_info(temp, tb_null)) mkdir(temp, S_IRWXU);
106 
107  // skip repeat '/'
108  while (*p && *p == '/') p++;
109  }
110  else p++;
111  }
112 
113  // open it again
114  fd = open(path, flags, modes);
115  }
116 
117  // ok?
118  return tb_fd2handle(fd);
119 }

194 tb_hong_t tb_file_writf(tb_handle_t file, tb_handle_t ifile, tb_hize_t offset, tb_hize_t size)
195 {
196  // check
197  tb_assert_and_check_return_val(file && ifile && size, -1);
198 
199 #if defined(TB_CONFIG_OS_LINUX) || defined(TB_CONFIG_OS_ANDROID)
200 
201  // writ it
202  off_t seek = offset;
203  tb_hong_t real = sendfile(tb_handle2fd(file), tb_handle2fd(ifile), &seek, (size_t)size);
204 
205  // ok?
206  if (real >= 0) return real;
207 
208  // continue?
209  if (errno == EINTR || errno == EAGAIN) return 0;
210 
211  // error
212  return -1;
213 
214 #else
215 
216  // read data
217  tb_byte_t data[8192];
218  tb_long_t read = tb_file_pread(ifile, data, sizeof(data), offset);
219  tb_check_return_val(read > 0, read);
220 
221  // writ data
222  tb_size_t writ = 0;
223  while (writ < read)
224  {
225  tb_long_t real = tb_file_writ(file, data + writ, read - writ);
226  if (real > 0) writ += real;
227  else break;
228  }
229 
230  // ok?
231  return writ == read? writ : -1;
232 #endif
233 }
