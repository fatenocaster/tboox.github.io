95 
96  // identification number
97  tb_static_stream_writ_u16_be(&sstream, TB_DNS_HEADER_MAGIC);
98 
99  /* 0x2104: 0 0000 001 0000 0000
100  *
101  * tb_uint16_t qr :1; // query/response flag
102  * tb_uint16_t opcode :4; // purpose of message
103  * tb_uint16_t aa :1; // authoritive answer
104  * tb_uint16_t tc :1; // truncated message
105  * tb_uint16_t rd :1; // recursion desired
106 
107  * tb_uint16_t ra :1; // recursion available
108  * tb_uint16_t z :1; // its z! reserved
109  * tb_uint16_t ad :1; // authenticated data
110  * tb_uint16_t cd :1; // checking disabled
111  * tb_uint16_t rcode :4; // response code
112  *
113  * this is a query
114  * this is a standard query
115  * not authoritive answer
116  * not truncated
117  * recursion desired
118  *
119  * recursion not available! hey we dont have it (lol)
120  *
121  */
122 #if 1
123  tb_static_stream_writ_u16_be(&sstream, 0x0100);
124 #else
125  tb_static_stream_writ_u1(&sstream, 0); // this is a query
126  tb_static_stream_writ_ubits32(&sstream, 0, 4); // this is a standard query
127  tb_static_stream_writ_u1(&sstream, 0); // not authoritive answer
128  tb_static_stream_writ_u1(&sstream, 0); // not truncated
129  tb_static_stream_writ_u1(&sstream, 1); // recursion desired
130 
131  tb_static_stream_writ_u1(&sstream, 0); // recursion not available! hey we dont have it (lol)
132  tb_static_stream_writ_u1(&sstream, 0);
133  tb_static_stream_writ_u1(&sstream, 0);
134  tb_static_stream_writ_u1(&sstream, 0);
135  tb_static_stream_writ_ubits32(&sstream, 0, 4);
136 #endif
137 
138  /* we have only one question
139  *
140  * tb_uint16_t question; // number of question entries
141  * tb_uint16_t answer; // number of answer entries
142  * tb_uint16_t authority; // number of authority entries
143  * tb_uint16_t resource; // number of resource entries
144  *
145  */
146  tb_static_stream_writ_u16_be(&sstream, 1);
147  tb_static_stream_writ_u16_be(&sstream, 0);
148  tb_static_stream_writ_u16_be(&sstream, 0);
149  tb_static_stream_writ_u16_be(&sstream, 0);

166 static tb_bool_t tb_aicp_dns_resp_done(tb_aicp_dns_impl_t* impl, tb_size_t size, tb_ipv4_t* ipv4)

191 
192  // check header

207 #endif
208 
209  // decode answers
210  tb_size_t i = 0;
211  tb_size_t found = 0;
212  for (i = 0; i < header.answer; i++)
213  {
214  // decode answer
215  tb_dns_answer_t answer;
216  tb_trace_d("response: answer: %d", i);
217 
218  // decode impl name

220  tb_trace_d("response: name: %s", name);
221 
222  // decode resource

229  tb_trace_d("response: ttl: %d", answer.res.ttl);
230  tb_trace_d("response: size: %d", answer.res.size);
231 
232  // is ipv4?
233  if (answer.res.type == 1)
234  {
235  tb_byte_t b1 = tb_static_stream_read_u8(&sstream);
236  tb_byte_t b2 = tb_static_stream_read_u8(&sstream);
237  tb_byte_t b3 = tb_static_stream_read_u8(&sstream);
238  tb_byte_t b4 = tb_static_stream_read_u8(&sstream);
239  tb_trace_d("response: ipv4: %u.%u.%u.%u", b1, b2, b3, b4);
240 
241  // save the first ip
242  if (!found)
243  {
244  // save it
245  if (ipv4)
246  {
247  ipv4->u8[0] = b1;
248  ipv4->u8[1] = b2;
249  ipv4->u8[2] = b3;
250  ipv4->u8[3] = b4;
251  }
252 
253  // found it
254  found = 1;
255  tb_trace_d("response: ");
256  break;
257  }
258  }
259  else
260  {
261  // decode rdata
262  answer.rdata = (tb_byte_t const*)tb_dns_decode_name(&sstream, answer.name);
263  tb_trace_d("response: alias: %s", answer.rdata? (tb_char_t const*)answer.rdata : "");
264  }
265  tb_trace_d("response: ");
266  }
267 
268  // found it?
269  tb_check_return_val(found, tb_false);
270 
271  // ok
272  return tb_true;
273 }
