411  else if (!tb_strnicmp(p, "file://", 7)) type = TB_STREAM_TYPE_FILE;

415  else if (!tb_strstr(p, "://")) type = TB_STREAM_TYPE_FILE;
416  else
417  {
418  tb_trace_d("[astream]: unknown prefix for url: %s", url);
419  return tb_null;
420  }
421  tb_assert_and_check_goto(type && type < tb_arrayn(g_init) && g_init[type], fail);
422 
423  // init stream

425  tb_assert_and_check_goto(astream, fail);
426

429 
430  // ok

432 
433 fail:
