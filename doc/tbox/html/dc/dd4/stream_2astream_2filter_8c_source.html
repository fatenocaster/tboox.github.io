176 static tb_bool_t tb_astream_filter_read_func(tb_astream_t* astream, tb_size_t state, tb_byte_t const* data, tb_size_t real, tb_size_t size, tb_pointer_t priv)
177 {
178  // check
179  tb_assert_and_check_return_val(astream, tb_false);
180 
181  // the stream
182  tb_astream_filter_t* fstream = (tb_astream_filter_t*)priv;
183  tb_assert_and_check_return_val(fstream && fstream->func.read, tb_false);
184 
185  // done filter
186  tb_bool_t ok = tb_false;
187  if (fstream->filter)
188  {
189  // done filter
190  switch (state)
191  {
192  case TB_STATE_OK:
193  {
194  // spak the filter
195  tb_long_t spak = tb_filter_spak(fstream->filter, data, real, &data, size, 0);
196 
197  // has output data?
198  if (spak > 0 && data)
199  {
200  // save offset
201  tb_atomic64_fetch_and_add(&fstream->offset, spak);
202 
203  // done data
204  ok = fstream->func.read((tb_astream_t*)fstream, TB_STATE_OK, data, spak, size, fstream->priv);
205  }
206  // no data? continue it
207  else if (!spak) ok = tb_true;
208  // closed?
209  else
210  {
211  // done closed
212  fstream->func.read((tb_astream_t*)fstream, TB_STATE_CLOSED, tb_null, 0, size, fstream->priv);
213 
214  // break it
215  // ok = tb_false;
216  }
217 
218  // eof and continue?
219  if (tb_filter_beof(fstream->filter) && ok)
220  {
221  // done sync for reading
222  ok = tb_astream_task(fstream->astream, 0, tb_astream_filter_sync_read_func, fstream);
223 
224  // error? done error
225  if (!ok) fstream->func.read((tb_astream_t*)fstream, TB_STATE_UNKNOWN_ERROR, tb_null, 0, size, fstream->priv);
226 
227  // need not read data, break it
228  ok = tb_false;
229  }
230  }
231  break;
232  case TB_STATE_CLOSED:
233  {
234  // done sync for reading
235  ok = tb_astream_task(fstream->astream, 0, tb_astream_filter_sync_read_func, fstream);
236 
237  // error? done error
238  if (!ok) fstream->func.read((tb_astream_t*)fstream, TB_STATE_UNKNOWN_ERROR, tb_null, 0, size, fstream->priv);
239  }
240  break;
241  default:
242  {
243  // done closed or failed
244  fstream->func.read((tb_astream_t*)fstream, state, tb_null, 0, size, fstream->priv);
245 
246  // break it
247  // ok = tb_false;
248  }
249  break;
250  }
251  }
252  // done func
253  else ok = fstream->func.read((tb_astream_t*)fstream, state, data, real, size, fstream->priv);
254 
255  // ok?
256  return ok;
