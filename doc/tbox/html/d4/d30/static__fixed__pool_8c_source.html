377 {

386

397  tb_size_t patch = 1;
398 #else
399  tb_size_t patch = 0;
400 #endif
401 
402  // init the item space
403  impl->item_space = impl->data_head_size + item_size + patch;
404  impl->item_space = tb_align(impl->item_space, TB_POOL_DATA_ALIGN);

410

412  *

416  * (maxn / 8) + (7 / 8) + maxn * item_space < left
417  * maxn * (1 / 8 + item_space) < left - (7 / 8)
