101 static tb_char_t const* tb_xml_reader_element_parse(tb_xml_reader_impl_t* reader)
102 {
103  // clear element
104  tb_string_clear(&reader->element);
105 
106  // parse element
107  tb_char_t ch = '\0';
108  tb_size_t in = 0;
109  while ((ch = tb_stream_bread_s8(reader->rstream)))
110  {
111  // append element
112  if (!in && ch == '<') in = 1;
113  else if (in)
114  {
115  if (ch != '>') tb_string_chrcat(&reader->element, ch);
116  else return tb_string_cstr(&reader->element);
117  }
118  }
119 
120  // failed
121  tb_assertf_abort(0, "invalid element: %s from %s", tb_string_cstr(&reader->element), tb_url_get(tb_stream_url(reader->istream)));
122  return tb_null;
123 }

344 tb_size_t tb_xml_reader_next(tb_xml_reader_ref_t reader)
345 {
346  // check
347  tb_xml_reader_impl_t* impl = (tb_xml_reader_impl_t*)reader;
348  tb_assert_and_check_return_val(impl && impl->rstream, TB_XML_READER_EVENT_NONE);
349 
350  // reset event
351  impl->event = TB_XML_READER_EVENT_NONE;
352 
353  // next
354  while (!impl->event)
355  {
356  // peek character
357  tb_char_t* pc = tb_null;
358  if (!tb_stream_need(impl->rstream, (tb_byte_t**)&pc, 1) || !pc) break;
359 
360  // is element?
361  if (*pc == '<')
362  {
363  // parse element: <...>
364  tb_char_t const* element = tb_xml_reader_element_parse(impl);
365  tb_assert_and_check_break(element);
366 
367  // is document begin: <?xml version="..." charset=".." ?>
368  tb_size_t size = tb_string_size(&impl->element);
369  if (size > 4 && !tb_strnicmp(element, "?xml", 4))
370  {
371  // update event
372  impl->event = TB_XML_READER_EVENT_DOCUMENT;
373 
374  // update version & charset
375  tb_xml_node_ref_t attr = (tb_xml_node_ref_t)tb_xml_reader_attributes(reader);
376  for (; attr; attr = attr->next)
377  {
378  if (!tb_string_cstricmp(&attr->name, "version")) tb_string_strcpy(&impl->version, &attr->data);
379  if (!tb_string_cstricmp(&attr->name, "encoding")) tb_string_strcpy(&impl->charset, &attr->data);
380  }
381 
382  // transform stream => utf-8
383  if (tb_string_cstricmp(&impl->charset, "utf-8") && tb_string_cstricmp(&impl->charset, "utf8"))
384  {
385  // charset
386  tb_size_t charset = TB_CHARSET_TYPE_UTF8;
387  if (!tb_string_cstricmp(&impl->charset, "gb2312") || !tb_string_cstricmp(&impl->charset, "gbk"))
388  charset = TB_CHARSET_TYPE_GB2312;
389  else tb_trace_e("the charset: %s is not supported", tb_string_cstr(&impl->charset));
390 
391  // init transform stream
392  if (charset != TB_CHARSET_TYPE_UTF8)
393  {
394 #ifdef TB_CONFIG_MODULE_HAVE_CHARSET
395  // init the filter stream
396  if (!impl->fstream) impl->fstream = tb_stream_init_filter_from_charset(impl->istream, charset, TB_CHARSET_TYPE_UTF8);
397  else
398  {
399  // ctrl stream
400  if (!tb_stream_ctrl(impl->fstream, TB_STREAM_CTRL_FLTR_SET_STREAM, impl->istream)) break;
401 
402  // the filter
403  tb_stream_filter_ref_t filter = tb_null;
404  if (!tb_stream_ctrl(impl->fstream, TB_STREAM_CTRL_FLTR_GET_FILTER, &filter)) break;
405  tb_assert_and_check_break(filter);
406 
407  // ctrl filter
408  if (!tb_stream_filter_ctrl(filter, TB_STREAM_FILTER_CTRL_CHARSET_SET_FTYPE, charset)) break;
409  }
410 
411  // open the filter stream
412  if (impl->fstream && tb_stream_open(impl->fstream))
413  impl->rstream = impl->fstream;
414  tb_string_cstrcpy(&impl->charset, "utf-8");
415 #else
416  // trace
417  tb_trace_e("unicode type is not supported, please enable charset module config if you want to use it!");
418 #endif
419  }
420  }
421  }
422  // is document type: <!DOCTYPE ... >
423  else if (size > 8 && !tb_strnicmp(element, "!DOCTYPE", 8))
424  {
425  // update event
426  impl->event = TB_XML_READER_EVENT_DOCUMENT_TYPE;
427  }
428  // is element end: </name>
429  else if (size > 1 && element[0] == '/')
430  {
431  // check
432  tb_check_break(impl->level);
433 
434  // update event
435  impl->event = TB_XML_READER_EVENT_ELEMENT_END;
436 
437  // leave
438  impl->level--;
439  }
440  // is comment: <!-- text -->
441  else if (size >= 3 && !tb_strncmp(element, "!--", 3))
442  {
443  // no comment end?
444  if (element[size - 2] != '-' || element[size - 1] != '-')
445  {
446  // patch '>'
447  tb_string_chrcat(&impl->element, '>');
448 
449  // seek to comment end
450  tb_char_t ch = '\0';
451  tb_int_t n = 0;
452  while ((ch = tb_stream_bread_s8(impl->rstream)))
453  {
454  // -->
455  if (n == 2 && ch == '>') break;
456  else
457  {
458  // append it
459  tb_string_chrcat(&impl->element, ch);
460 
461  if (ch == '-') n++;
462  else n = 0;
463  }
464  }
465 
466  // update event
467  if (ch != '\0') impl->event = TB_XML_READER_EVENT_COMMENT;
468  }
469  else impl->event = TB_XML_READER_EVENT_COMMENT;
470  }
471  // is cdata: <![CDATA[ text ]]>
472  else if (size >= 8 && !tb_strnicmp(element, "![CDATA[", 8))
473  {
474  if (element[size - 2] != ']' || element[size - 1] != ']')
475  {
476  // patch '>'
477  tb_string_chrcat(&impl->element, '>');
478 
479  // seek to cdata end
480  tb_char_t ch = '\0';
481  tb_int_t n = 0;
482  while ((ch = tb_stream_bread_s8(impl->rstream)))
483  {
484  // ]]>
485  if (n == 2 && ch == '>') break;
486  else
487  {
488  // append it
489  tb_string_chrcat(&impl->element, ch);
490 
491  if (ch == ']') n++;
492  else n = 0;
493  }
494  }
495 
496  // update event
497  if (ch != '\0') impl->event = TB_XML_READER_EVENT_CDATA;
498  }
499  else impl->event = TB_XML_READER_EVENT_CDATA;
500  }
501  // is empty element: <name/>
502  else if (size > 1 && element[size - 1] == '/')
503  {
504  // update event
505  impl->event = TB_XML_READER_EVENT_ELEMENT_EMPTY;
506  }
507  // is element begin: <name>
508  else
509  {
510  // update event
511  impl->event = TB_XML_READER_EVENT_ELEMENT_BEG;
512 
513  // enter
514  impl->level++;
515  }
516 
517  // trace
518 // tb_trace_d("<%s>", element);
519  }
520  // is text: <> text </>
521  else if (*pc)
522  {
523  // parse text: <> ... <>
524  tb_char_t const* text = tb_xml_reader_text_parse(impl);
525  if (text && tb_string_cstrcmp(&impl->text, "\r\n") && tb_string_cstrcmp(&impl->text, "\n"))
526  impl->event = TB_XML_READER_EVENT_TEXT;
527 
528  // trace
529 // tb_trace_d("%s", text);
530  }
531  else
532  {
533  // skip the invalid character
534  if (!tb_stream_skip(impl->rstream, 1)) break;
535  }
536  }
537 
538  // ok?
539  return impl->event;
540 }
541 tb_bool_t tb_xml_reader_goto(tb_xml_reader_ref_t reader, tb_char_t const* path)
542 {
543  // check
544  tb_xml_reader_impl_t* impl = (tb_xml_reader_impl_t*)reader;
545  tb_assert_and_check_return_val(impl && impl->rstream && path, tb_false);
546  tb_trace_d("goto: %s", path);
547 
548  // init level
549  impl->level = 0;
550 
551  // seek to the stream head
552  if (!tb_stream_seek(impl->rstream, 0)) return tb_false;
553 
554  // init
555  tb_static_string_t s;
556  tb_char_t data[8192];
557  if (!tb_static_string_init(&s, data, 8192)) return tb_false;
558 
559  // save the current offset
560  tb_hize_t save = tb_stream_offset(impl->rstream);
561 
562  // done
563  tb_bool_t ok = tb_false;
564  tb_bool_t leave = tb_false;
565  tb_size_t event = TB_XML_READER_EVENT_NONE;
566  while (!leave && !ok && (event = tb_xml_reader_next(reader)))
567  {
568  switch (event)
569  {
570  case TB_XML_READER_EVENT_ELEMENT_EMPTY:
571  {
572  // name
573  tb_char_t const* name = tb_xml_reader_element(reader);
574  tb_assert_and_check_break_state(name, leave, tb_true);
575 
576  // append
577  tb_size_t n = tb_static_string_size(&s);
578  tb_static_string_chrcat(&s, '/');
579  tb_static_string_cstrcat(&s, name);
580 
581  // ok?
582  if (!tb_static_string_cstricmp(&s, path)) ok = tb_true;
583  tb_trace_d("path: %s", tb_static_string_cstr(&s));
584 
585  // remove
586  tb_static_string_strip(&s, n);
587 
588  // restore
589  if (ok) if (!(ok = tb_stream_seek(impl->rstream, save))) leave = tb_true;
590  }
591  break;
592  case TB_XML_READER_EVENT_ELEMENT_BEG:
593  {
594  // name
595  tb_char_t const* name = tb_xml_reader_element(reader);
596  tb_assert_and_check_break_state(name, leave, tb_true);
597 
598  // append
599  tb_static_string_chrcat(&s, '/');
600  tb_static_string_cstrcat(&s, name);
601 
602  // ok?
603  if (!tb_static_string_cstricmp(&s, path)) ok = tb_true;
604  tb_trace_d("path: %s", tb_static_string_cstr(&s));
605 
606  // restore
607  if (ok) if (!(ok = tb_stream_seek(impl->rstream, save))) leave = tb_true;
608  }
609  break;
610  case TB_XML_READER_EVENT_ELEMENT_END:
611  {
612  // remove
613  tb_long_t p = tb_static_string_strrchr(&s, 0, '/');
614  if (p >= 0) tb_static_string_strip(&s, p);
615 
616  // ok?
617  if (!tb_static_string_cstricmp(&s, path)) ok = tb_true;
618  tb_trace_d("path: %s", tb_static_string_cstr(&s));
619 
620  // restore
621  if (ok) if (!(ok = tb_stream_seek(impl->rstream, save))) leave = tb_true;
622  }
623  break;
624  default:
625  break;
626  }
627 
628  // save
629  save = tb_stream_offset(impl->rstream);
630  }
631 
632  // exit string
633  tb_static_string_exit(&s);
634 
635  // clear level
636  impl->level = 0;
637 
638  // failed? restore to the stream head
639  if (!ok) tb_stream_seek(impl->rstream, 0);
640 
641  // ok?
642  return ok;
643 }
644 tb_xml_node_ref_t tb_xml_reader_load(tb_xml_reader_ref_t reader)
645 {
646  // check
647  tb_assert_and_check_return_val(reader, tb_null);
648 
649  // done
650  tb_bool_t ok = tb_true;
651  tb_xml_node_ref_t node = tb_null;
652  tb_size_t event = TB_XML_READER_EVENT_NONE;
653  while (ok && (event = tb_xml_reader_next(reader)))
654  {
655  // init document node
656  if (!node)
657  {
658  node = tb_xml_node_init_document(tb_xml_reader_version(reader), tb_xml_reader_charset(reader));
659  tb_assert_and_check_break_state(node && !node->parent, ok, tb_false);
660  }
661 
662  switch (event)
663  {
664  case TB_XML_READER_EVENT_DOCUMENT:
665  break;
666  case TB_XML_READER_EVENT_DOCUMENT_TYPE:
667  {
668  // init
669  tb_xml_node_ref_t doctype = tb_xml_node_init_document_type(tb_xml_reader_doctype(reader));
670  tb_assert_and_check_break_state(doctype, ok, tb_false);
671 
672  // append
673  tb_xml_node_append_ctail(node, doctype);
674  tb_assert_and_check_break_state(doctype->parent, ok, tb_false);
675  }
676  break;
677  case TB_XML_READER_EVENT_ELEMENT_EMPTY:
678  {
679  // init
680  tb_xml_node_ref_t element = tb_xml_node_init_element(tb_xml_reader_element(reader));
681  tb_assert_and_check_break_state(element, ok, tb_false);
682 
683  // attributes
684  tb_xml_node_ref_t attr = tb_xml_reader_attributes(reader);
685  for (; attr; attr = attr->next)
686  tb_xml_node_append_atail(element, tb_xml_node_init_attribute(tb_string_cstr(&attr->name), tb_string_cstr(&attr->data)));
687 
688  // append
689  tb_xml_node_append_ctail(node, element);
690  tb_assert_and_check_break_state(element->parent, ok, tb_false);
691  }
692  break;
693  case TB_XML_READER_EVENT_ELEMENT_BEG:
694  {
695  // init
696  tb_xml_node_ref_t element = tb_xml_node_init_element(tb_xml_reader_element(reader));
697  tb_assert_and_check_break_state(element, ok, tb_false);
698 
699  // attributes
700  tb_xml_node_ref_t attr = tb_xml_reader_attributes(reader);
701  for (; attr; attr = attr->next)
702  tb_xml_node_append_atail(element, tb_xml_node_init_attribute(tb_string_cstr(&attr->name), tb_string_cstr(&attr->data)));
703 
704  // append
705  tb_xml_node_append_ctail(node, element);
706  tb_assert_and_check_break_state(element->parent, ok, tb_false);
707 
708  // enter
709  node = element;
710  }
711  break;
712  case TB_XML_READER_EVENT_ELEMENT_END:
713  {
714  // check
715  tb_assert_and_check_break_state(node, ok, tb_false);
716 
717  // the parent node
718  node = node->parent;
719  }
