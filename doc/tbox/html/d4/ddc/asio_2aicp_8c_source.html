431 tb_void_t tb_aicp_loop_util(tb_aicp_ref_t aicp, tb_bool_t (*stop)(tb_cpointer_t priv), tb_cpointer_t priv)
432 {
433  // check
434  tb_aicp_impl_t* impl = (tb_aicp_impl_t*)aicp;
435  tb_assert_and_check_return(impl);
436 
437  // the ptor
438  tb_aicp_ptor_impl_t* ptor = impl->ptor;
439  tb_assert_and_check_return(ptor && ptor->loop_spak);
440 
441  // the loop spak
442  tb_long_t (*loop_spak)(tb_aicp_ptor_impl_t* , tb_handle_t, tb_aice_t* , tb_long_t ) = ptor->loop_spak;
443 
444  // worker++
445  tb_atomic_fetch_and_inc(&impl->work);
446 
447  // init loop
448  tb_handle_t loop = ptor->loop_init? ptor->loop_init(ptor) : tb_null;
449 
450  // trace
451  tb_trace_d("loop[%p]: init", loop);
452 
453  // spak ctime
454  tb_cache_time_spak();
455 
456  // loop
457  while (1)
458  {
459  // spak
460  tb_aice_t resp = {0};
461  tb_long_t ok = loop_spak(ptor, loop, &resp, -1);
462 
463  // spak ctime
464  tb_cache_time_spak();
465 
466  // failed?
467  tb_check_break(ok >= 0);
468 
469  // timeout?
470  tb_check_continue(ok);
471 
472  // check aico
473  tb_aico_impl_t* aico = (tb_aico_impl_t*)resp.aico;
474  tb_assert_and_check_continue(aico);
475 
476  // trace
477  tb_trace_d("loop[%p]: spak: code: %lu, aico: %p, state: %s: %ld", loop, resp.code, aico, aico? tb_state_cstr(tb_atomic_get(&aico->state)) : "null", ok);
478 
479  // pending? clear state if be not accept or accept failed
480  tb_size_t state = TB_STATE_OPENED;
481  state = (resp.code != TB_AICE_CODE_ACPT || resp.state != TB_STATE_OK)? tb_atomic_fetch_and_pset(&aico->state, TB_STATE_PENDING, state) : tb_atomic_get(&aico->state);
482 
483  // killed or killing?
484  if (state == TB_STATE_KILLED || state == TB_STATE_KILLING)
485  {
486  // update the aice state
487  resp.state = TB_STATE_KILLED;
488 
489  // killing? update to the killed state
490  tb_atomic_fetch_and_pset(&aico->state, TB_STATE_KILLING, TB_STATE_KILLED);
491  }
492 
493  // done func, @note maybe the aico exit will be called
494  if (resp.func && !resp.func(&resp))
495  {
496  // trace
497 #ifdef __tb_debug__
498  tb_trace_e("loop[%p]: done aice func failed with code: %lu at line: %lu, func: %s, file: %s!", loop, resp.code, aico->line, aico->func, aico->file);
499 #else
500  tb_trace_e("loop[%p]: done aice func failed with code: %lu!", loop, resp.code);
501 #endif
502  }
503 
504  // killing? update to the killed state
505  tb_atomic_fetch_and_pset(&aico->state, TB_STATE_KILLING, TB_STATE_KILLED);
506 
507  // stop it?
508  if (stop && stop(priv)) tb_aicp_kill(aicp);
509  }
510 
511  // exit loop
512  if (ptor->loop_exit) ptor->loop_exit(ptor, loop);
513 
514  // worker--
515  tb_atomic_fetch_and_dec(&impl->work);
516 
517  // trace
518  tb_trace_d("loop[%p]: exit", loop);
519 }
