36 tb_file_ref_t tb_file_init(tb_char_t const* path, tb_size_t mode)
37 {
38  // check
39  tb_assert_and_check_return_val(path, tb_null);
40 
41  // the full path
42  tb_wchar_t full[TB_PATH_MAXN];
43  if (!tb_path_full_w(path, full, TB_PATH_MAXN)) return tb_null;
44 
45  // init access
46  DWORD access = GENERIC_READ;
47  if (mode & TB_FILE_MODE_RO) access = GENERIC_READ;
48  else if (mode & TB_FILE_MODE_WO) access = GENERIC_WRITE;
49  else if (mode & TB_FILE_MODE_RW) access = GENERIC_READ | GENERIC_WRITE;
50 
51  // init share
52  DWORD share = FILE_SHARE_READ;
53  if (mode & TB_FILE_MODE_RO) share = FILE_SHARE_READ;
54  else if (mode & TB_FILE_MODE_WO) share = FILE_SHARE_WRITE;
55  else if (mode & TB_FILE_MODE_RW) share = FILE_SHARE_READ | FILE_SHARE_WRITE;
56 
57  // init flag
58  DWORD cflag = 0;
59  if (mode & (TB_FILE_MODE_CREAT | TB_FILE_MODE_TRUNC)) cflag |= CREATE_ALWAYS;
60  else if (mode & TB_FILE_MODE_CREAT) cflag |= CREATE_NEW;
61  else if (mode & TB_FILE_MODE_TRUNC) cflag |= TRUNCATE_EXISTING;
62  if (!cflag) cflag |= OPEN_EXISTING;
63 
64  // init attr
65  DWORD attr = FILE_ATTRIBUTE_NORMAL;
66  if (mode & TB_FILE_MODE_ASIO) attr |= FILE_FLAG_OVERLAPPED;
67  if (mode & TB_FILE_MODE_DIRECT) attr |= FILE_FLAG_NO_BUFFERING;
68 
69  // init file
70  HANDLE file = CreateFileW(full, access, share, tb_null, cflag, attr, tb_null);
71  if (file == INVALID_HANDLE_VALUE && (mode & TB_FILE_MODE_CREAT))
72  {
73  // make directory
74  tb_wchar_t temp[TB_PATH_MAXN] = {0};
75  tb_wchar_t const* p = full;
76  tb_wchar_t* t = temp;
77  tb_wchar_t const* e = temp + TB_PATH_MAXN - 1;
78  for (; t < e && *p; t++)
79  {
80  *t = *p;
81  if (*p == L'\\' || *p == L'/')
82  {
83  // make directory if not exists
84  if (INVALID_FILE_ATTRIBUTES == GetFileAttributesW(temp)) CreateDirectoryW(temp, tb_null);
85 
86  // skip repeat '\\' or '/'
87  while (*p && (*p == L'\\' || *p == L'/')) p++;
88  }
89  else p++;
90  }
91 
92  // init it again
93  file = CreateFileW(full, access, share, tb_null, cflag, attr, tb_null);
94  }
95 
96  // append?
97  if (file != INVALID_HANDLE_VALUE && (mode & TB_FILE_MODE_APPEND))
98  {
99  // seek to end
100  tb_hize_t size = tb_file_size((tb_file_ref_t)file);
101  if (size)
102  {
103  if (size != tb_file_seek((tb_file_ref_t)file, size, TB_FILE_SEEK_BEG))
104  {
105  tb_file_exit((tb_file_ref_t)file);
106  file = INVALID_HANDLE_VALUE;
107  }
108  }
109  }
110 
111  // ok?
112  return file != INVALID_HANDLE_VALUE? (tb_file_ref_t)file : tb_null;
113 }

192 
193  // walk read

195  tb_size_t read = 0;
196  for (i = 0; i < size; i++)
197  {
198  // the data & size

202

205 
206  // full? next it
207  if (real == need)

209  read += real;
210  continue ;
211  }
212 
213  // failed?

215 
216  // ok?

218 
219  // end
220  break;
221  }
222 
223  // ok?
224  return read;

264 tb_hong_t tb_file_writf(tb_file_ref_t file, tb_file_ref_t ifile, tb_hize_t offset, tb_hize_t size)
265 {
266  // check
267  tb_assert_and_check_return_val(file && ifile && size, -1);
268 
269  // read data
270  tb_byte_t data[8192];
271  tb_long_t read = tb_file_pread(ifile, data, sizeof(data), offset);
272  tb_check_return_val(read > 0, read);
273 
274  // writ data
275  tb_long_t writ = 0;
276  while (writ < read)
277  {
278  tb_long_t real = tb_file_writ(file, data + writ, read - writ);
279  if (real > 0) writ += real;
280  else break;
281  }
282 
283  // ok?
284  return writ == read? writ : -1;
285 }
