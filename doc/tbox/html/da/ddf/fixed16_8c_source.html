222 tb_fixed16_t tb_fixed16_invert_int32(tb_fixed16_t x)
223 {
224  // is one?
225  if (x == TB_FIXED16_ONE) return TB_FIXED16_ONE;
226 
227  // get sign
228  tb_int32_t s = tb_int32_get_sign(x);
229 
230  // abs(x)
231  x = tb_fixed16_abs(x);
232 
233  // is infinity?
234  if (x <= 2) return tb_int32_set_sign(TB_FIXED16_MAX, s);
235 
236  // normalize
237  tb_int32_t cl0 = tb_bits_cl0_u32_be(x);
238  x = x << cl0 >> 16;
239 
240  // compute 1 / x approximation (0.5 <= x < 1.0)
241  // (2.90625 (~2.914) - 2 * x) >> 1
242  tb_uint32_t r = 0x17400 - x;
243 
244  // newton-raphson iteration:
245  // x = r * (2 - x * r) = ((r / 2) * (1 - x * r / 2)) * 4
246  r = ((0x10000 - ((x * r) >> 16)) * r) >> 15;
247  r = ((0x10000 - ((x * r) >> 16)) * r) >> (30 - cl0);
248 
249  return tb_int32_set_sign(r, s);
250 }
