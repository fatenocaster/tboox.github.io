103 static tb_long_t tb_cstream_spak(tb_gstream_t* gst, tb_long_t sync)
104 {
105  tb_cstream_t* cst = tb_cstream_cast(gst);
106  tb_tstream_t* tst = tb_tstream_cast(gst);
107  tb_assert_and_check_return_val(cst && tst, -1);
108 
109  // the convecter
110  tb_assert_and_check_return_val(TB_CHARSET_TYPE_OK(cst->ftype) && TB_CHARSET_TYPE_OK(cst->ttype), -1);
111 
112  // the input
113  tb_assert_and_check_return_val(tst->ip, -1);
114  tb_byte_t const* ib = tst->ip;
115  tb_byte_t const* ip = tst->ip;
116  tb_byte_t const* ie = ip + tst->in;
117  tb_check_return_val(ip < ie, 0);
118 
119  // the output
120  tb_assert_and_check_return_val(tst->op, -1);
121  tb_byte_t* ob = tst->op;
122  tb_byte_t* op = tst->op;
123  tb_byte_t const* oe = tst->ob + TB_TSTREAM_CACHE_MAXN;
124  tb_check_return_val(op < oe, 0);
125 
126  // spak it
127  tb_bstream_t ist;
128  tb_bstream_t ost;
129  tb_bstream_init(&ist, ib, ie - ib);
130  tb_bstream_init(&ost, ob, oe - ob);
131  if (tb_charset_conv_bst(cst->ftype, cst->ttype, &ist, &ost) < 0) return -1;
132  ip = tb_bstream_pos(&ist);
133  op = tb_bstream_pos(&ost);
134 
135  // check
136  tb_assert_and_check_return_val(ip >= ib && ip <= ie, -1);
137  tb_assert_and_check_return_val(op >= ob && op <= oe, -1);
138 
139  // update input
140  tst->in -= ip - ib;
141  tst->ip = ip;
142 
143  // update output
144  tst->on += op - ob;
145 
146  // ok
147  return (op - ob);
148 }
