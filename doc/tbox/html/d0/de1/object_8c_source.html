175 tb_object_ref_t tb_object_seek(tb_object_ref_t object, tb_char_t const* path, tb_size_t type)
176 {
177  // check
178  tb_assert_and_check_return_val(object, tb_null);
179 
180  // null?
181  tb_check_return_val(path, object);
182 
183  // walk
184  tb_char_t const* p = path;
185  tb_char_t const* e = path + tb_strlen(path);
186  while (p < e && object)
187  {
188  // done seek
189  switch (*p)
190  {
191  case '.':
192  {
193  // check
194  tb_assert_and_check_return_val(tb_object_type(object) == TB_OBJECT_TYPE_DICTIONARY, tb_null);
195 
196  // skip
197  p++;
198 
199  // read the key name
200  tb_char_t key[4096] = {0};
201  tb_char_t* kb = key;
202  tb_char_t* ke = key + 4095;
203  for (; p < e && kb < ke && *p && (*p != '.' && *p != '[' && *p != ']'); p++, kb++)
204  {
205  if (*p == '\\') p++;
206  *kb = *p;
207  }
208 
209  // trace
210  tb_trace_d("key: %s", key);
211 
212  // the value
213  object = tb_object_dictionary_val(object, key);
214  }
215  break;
216  case '[':
217  {
218  // check
219  tb_assert_and_check_return_val(tb_object_type(object) == TB_OBJECT_TYPE_ARRAY, tb_null);
220 
221  // skip
222  p++;
223 
224  // read the item index
225  tb_char_t index[32] = {0};
226  tb_char_t* ib = index;
227  tb_char_t* ie = index + 31;
228  for (; p < e && ib < ie && *p && tb_isdigit10(*p); p++, ib++) *ib = *p;
229 
230  // trace

232 
233  // check

235  tb_assert_and_check_return_val(i < tb_object_array_size(object), tb_null);
236 
237  // the value
238  object = tb_object_array_item(object, i);
239  }
240  break;
241  case ']':
242  default:
243  p++;
244  break;
245  }
246  }
247 
248  // check it, if not none
249  if (object && type != TB_OBJECT_TYPE_NONE)
250  {
251  // is this type?

254
