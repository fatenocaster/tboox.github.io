99 static tb_bool_t tb_gstream_sock_open(tb_handle_t gstream)
100 {
101  // check
102  tb_gstream_sock_t* sstream = tb_gstream_sock_cast(gstream);
103  tb_assert_and_check_return_val(sstream && sstream->type, tb_false);
104 
105  // clear
106  sstream->wait = 0;
107  sstream->tryn = 0;
108  sstream->read = 0;
109  sstream->writ = 0;
110 
111  // opened?
112  tb_check_return_val(!sstream->sock, tb_true);
113 
114  // port
115  tb_size_t port = tb_url_port_get(&sstream->base.base.url);
116  tb_assert_and_check_return_val(port, tb_false);
117 
118  // ipv4
119  if (!sstream->addr.u32)
120  {
121  // try to get the ipv4 address from url
122  tb_ipv4_t const* ipv4 = tb_url_ipv4_get(&sstream->base.base.url);
123  if (ipv4 && ipv4->u32) sstream->addr = *ipv4;
124  else
125  {
126  // look addr
127  if (!tb_dns_looker_done(tb_url_host_get(&sstream->base.base.url), &sstream->addr))
128  {
129  sstream->base.state = TB_STATE_SOCK_DNS_FAILED;
130  return tb_false;
131  }
132 
133  // save addr
134  tb_url_ipv4_set(&sstream->base.base.url, &sstream->addr);
135  }
136 
137  // tcp or udp? for url: sock://ip:port/?udp=
138  tb_char_t const* args = tb_url_args_get(&sstream->base.base.url);
139  if (args && !tb_strnicmp(args, "udp=", 4)) sstream->type = TB_SOCKET_TYPE_UDP;
140  else if (args && !tb_strnicmp(args, "tcp=", 4)) sstream->type = TB_SOCKET_TYPE_TCP;
141  }
142 
143  // make sock
144  sstream->sock = tb_socket_open(sstream->type);
145  sstream->bref = 0;
146  tb_assert_and_check_return_val(sstream->sock, tb_false);
147 
148  // resize cache
149  tb_size_t recv_cache = tb_socket_recv_buffer_size(sstream->sock);
150  tb_size_t send_cache = tb_socket_send_buffer_size(sstream->sock);
151  if (recv_cache || send_cache) tb_qbuffer_resize(&sstream->base.cache, tb_max(recv_cache, send_cache));
152 
153  // done
154  tb_bool_t ok = tb_false;
155  switch (sstream->type)
156  {
157  case TB_SOCKET_TYPE_TCP:
158  {
159  // trace
160  tb_trace_d("connect: %s[%u.%u.%u.%u]:%u: ..", tb_url_host_get(&sstream->base.base.url), tb_ipv4_u8x4(sstream->addr), port);
161 
162  // connect it
163  tb_long_t real = -1;
164  while (!(real = tb_socket_connect(sstream->sock, &sstream->addr, port)) && !tb_atomic_get(&sstream->base.base.bstoped))
165  {
166  real = tb_aioo_wait(sstream->sock, TB_AIOE_CODE_CONN, tb_stream_timeout(gstream));
167  tb_check_break(real > 0);
168  }
169 
170  // ok?
171  if (real > 0)
172  {
173  ok = tb_true;
174  sstream->base.state = TB_STATE_OK;
175  }
176  else sstream->base.state = !real? TB_STATE_SOCK_CONNECT_TIMEOUT : TB_STATE_SOCK_CONNECT_FAILED;
177 
178  // trace
179  tb_trace_d("connect: %s", ok? "ok" : "no");
180 
181  // ok?
182  if (ok)
183  {
184  // ssl? init it
185  if (tb_url_ssl_get(&sstream->base.base.url))
186  {
187  // done
188  ok = tb_false;
189  do
190  {
191  // init ssl
192  if (!sstream->hssl) sstream->hssl = tb_ssl_init(tb_false);
193  tb_assert_and_check_break(sstream->hssl);
194 
195  // init bio
196  tb_ssl_set_bio_sock(sstream->hssl, sstream->sock);
197 
198  // init timeout
199  tb_ssl_set_timeout(sstream->hssl, tb_stream_timeout(gstream));
200 
201  // open ssl
202  if (!tb_ssl_open(sstream->hssl)) break;
203 
204  // ok
205  ok = tb_true;
206 
207  } while (0);
208 
209  // trace
210  tb_trace_d("ssl: %s", ok? "ok" : "no");
211 
212  // failed? save state
213  if (!ok) sstream->base.state = sstream->hssl? tb_ssl_state(sstream->hssl) : TB_STATE_SOCK_SSL_FAILED;
214  }
215  }
216  }
217  break;
218  case TB_SOCKET_TYPE_UDP:
219  {
220  // ssl? not supported
221  if (tb_url_ssl_get(&sstream->base.base.url))
222  {
223  // trace
224  tb_trace_w("udp ssl is not supported!");
225  }
226 
227  // ok
228  ok = tb_true;
229  sstream->base.state = TB_STATE_OK;
230  }
231  break;
232  default:
233  tb_trace_e("unknown socket type: %lu", sstream->type);
234  break;
235  }
236 
237  // ok?
238  return ok;
239 }
