178 static __tb_inline__ tb_fixed30_t tb_fixed30_mul_int32(tb_fixed30_t x, tb_fixed30_t y)
179 {
180  // get sign
181  tb_int32_t s = tb_int32_get_sign(x ^ y);
182  x = tb_fixed30_abs(x);
183  y = tb_fixed30_abs(y);
184 
185  tb_uint32_t xh = x >> 16;
186  tb_uint32_t xl = x & 0xffff;
187  tb_uint32_t yh = y >> 16;
188  tb_uint32_t yl = y & 0xffff;
189 
190  tb_uint32_t xyh = xh * yh;
191  tb_uint32_t xyl = xl * yl;
192  tb_uint32_t xyhl = xh * yl + xl * yh;
193

195  tb_uint32_t hi = xyh + (xyhl >> 16) + (lo < xyl);
