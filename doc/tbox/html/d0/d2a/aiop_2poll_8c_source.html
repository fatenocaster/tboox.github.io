213 static tb_long_t tb_aiop_reactor_poll_wait(tb_aiop_reactor_t* reactor, tb_aioe_t* list, tb_size_t maxn, tb_long_t timeout)
214 {
215  // check
216  tb_aiop_reactor_poll_t* rtor = (tb_aiop_reactor_poll_t*)reactor;
217  tb_assert_and_check_return_val(rtor && rtor->pfds && rtor->cfds && list && maxn, -1);
218 
219  // the aiop
220  tb_aiop_t* aiop = reactor->aiop;
221  tb_assert_and_check_return_val(aiop, tb_false);
222 
223  // loop
224  tb_long_t wait = 0;
225  tb_bool_t stop = tb_false;
226  tb_hong_t time = tb_mclock();
227  while (!wait && !stop && (timeout < 0 || tb_mclock() < time + timeout))
228  {
229  // copy pfds
230  if (rtor->mutx.pfds) tb_mutex_enter(rtor->mutx.pfds);
231  tb_vector_copy(rtor->cfds, rtor->pfds);
232  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
233 
234  // cfds
235  struct pollfd* cfds = (struct pollfd*)tb_vector_data(rtor->cfds);
236  tb_size_t cfdm = tb_vector_size(rtor->cfds);
237  tb_assert_and_check_return_val(cfds && cfdm, -1);
238 
239  // wait
240  tb_long_t cfdn = poll(cfds, cfdm, timeout);
241  tb_assert_and_check_return_val(cfdn >= 0, -1);
242 
243  // timeout?
244  tb_check_return_val(cfdn, 0);
245 
246  // sync
247  tb_size_t i = 0;
248  for (i = 0; i < cfdm && wait < maxn; i++)
249  {
250  // the handle
251  tb_handle_t handle = (tb_handle_t)(cfds[i].fd + 1);
252  tb_assert_and_check_return_val(handle, -1);
253 
254  // the events
255  tb_size_t events = cfds[i].revents;
256  tb_check_continue(events);
257 
258  // spak?
259  if (handle == aiop->spak[1] && (events & POLLIN))
260  {
261  // read spak
262  tb_char_t spak = '\0';
263  if (1 != tb_socket_recv(aiop->spak[1], &spak, 1)) return -1;
264 
265  // killed?
266  if (spak == 'k') return -1;
267 
268  // stop to wait
269  stop = tb_true;
270 
271  // continue it
272  continue ;
273  }
274 
275  // skip spak
276  tb_check_continue(handle != aiop->spak[1]);
277 
278  // the aioo
279  tb_size_t code = TB_AIOE_CODE_NONE;
280  tb_pointer_t data = tb_null;
281  tb_aioo_t* aioo = tb_null;
282  if (rtor->mutx.hash) tb_mutex_enter(rtor->mutx.hash);
283  if (rtor->hash)
284  {
285  aioo = (tb_aioo_t*)tb_hash_get(rtor->hash, handle);
286  if (aioo)
287  {
288  // save code & data
289  code = aioo->code;
290  data = aioo->data;
291 
292  // oneshot? clear it
293  if (aioo->code & TB_AIOE_CODE_ONESHOT)
294  {
295  aioo->code = TB_AIOE_CODE_NONE;
296  aioo->data = tb_null;
297  }
298  }
299  }
300  if (rtor->mutx.hash) tb_mutex_leave(rtor->mutx.hash);
301  tb_check_continue(aioo && code);
302 
303  // init aioe
304  tb_aioe_t aioe = {0};
305  aioe.data = data;
306  aioe.aioo = aioo;
307  if (events & POLLIN)
308  {
309  aioe.code |= TB_AIOE_CODE_RECV;
310  if (code & TB_AIOE_CODE_ACPT) aioe.code |= TB_AIOE_CODE_ACPT;
311  }
312  if (events & POLLOUT)
313  {
314  aioe.code |= TB_AIOE_CODE_SEND;
315  if (code & TB_AIOE_CODE_CONN) aioe.code |= TB_AIOE_CODE_CONN;
316  }
317  if ((events & POLLHUP) && !(code & (TB_AIOE_CODE_RECV | TB_AIOE_CODE_SEND)))
318  aioe.code |= TB_AIOE_CODE_RECV | TB_AIOE_CODE_SEND;
319 
320  // save aioe
321  list[wait++] = aioe;
322 
323  // oneshot?
324  if (code & TB_AIOE_CODE_ONESHOT)
325  {
326  if (rtor->mutx.pfds) tb_mutex_enter(rtor->mutx.pfds);
327  struct pollfd* pfds = (struct pollfd*)tb_vector_data(rtor->pfds);
328  if (pfds) pfds[i].events = 0;
329  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
330  }
331  }
332  }
333 
334  // ok
335  return wait;
336 }
