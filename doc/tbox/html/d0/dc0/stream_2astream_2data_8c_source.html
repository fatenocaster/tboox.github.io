164 static tb_bool_t tb_astream_data_read_func(tb_aice_t const* aice)
165 {
166  // check
167  tb_assert_and_check_return_val(aice && aice->aico && aice->code == TB_AICE_CODE_RUNTASK, tb_false);
168 
169  // the stream
170  tb_astream_data_t* dstream = (tb_astream_data_t*)aice->priv;
171  tb_assert_and_check_return_val(dstream && dstream->func.read.func, tb_false);
172 
173  // done state
174  tb_size_t state = TB_STATE_UNKNOWN_ERROR;
175  switch (aice->state)
176  {
177  // ok
178  case TB_STATE_OK:
179  {
180  // check
181  tb_assert_and_check_break(dstream->data && dstream->head);
182 
183  // the left
184  tb_size_t left = dstream->data + dstream->size - dstream->head;
185 
186  // the real
187  tb_size_t real = dstream->func.read.size;
188  if (!real || real > left) real = left;
189 
190  // no data? closed
191  if (!real)
192  {
193  state = TB_STATE_CLOSED;
194  break;
195  }
196 
197  // save data
198  tb_byte_t* data = dstream->head;
199 
200  // save head
201  dstream->head += real;
202 
203  // save offset
204  tb_atomic64_set(&dstream->offset, dstream->head - dstream->data);
205 
206  // ok
207  state = TB_STATE_OK;
208 
209  // done func
210  if (dstream->func.read.func((tb_astream_t*)dstream, state, data, real, dstream->func.read.size, dstream->func.read.priv))
211  {
212  // continue to post read
213  tb_aico_task_run(aice->aico, 0, tb_astream_data_read_func, (tb_astream_t*)dstream);
214  }
215  }
216  break;
217  // killed
218  case TB_STATE_KILLED:
219  state = TB_STATE_KILLED;
220  break;
221  default:
222  tb_trace_d("read: unknown state: %s", tb_state_cstr(aice->state));
223  break;
224  }
225 
226  // done func
227  if (state != TB_STATE_OK) dstream->func.read.func((tb_astream_t*)dstream, state, tb_null, 0, dstream->func.read.size, dstream->func.read.priv);
228 
229  // ok
230  return tb_true;
231 }

254 
255  // done state
256  tb_size_t state = TB_STATE_UNKNOWN_ERROR;
257  switch (aice->state)
258  {
259  // ok

262  // check
263  tb_assert_and_check_break(dstream->data && dstream->head);
264  tb_assert_and_check_break(dstream->func.writ.data && dstream->func.writ.size);

268

275  {
276  state = TB_STATE_CLOSED;
277  break;
278  }
279 
280  // save data
281  tb_memcpy(dstream->head, dstream->func.writ.data, real);
282 
283  // save head
284  dstream->head += real;
285 
286  // save offset
287  tb_atomic64_set(&dstream->offset, dstream->head - dstream->data);
288 
289  // ok
290  state = TB_STATE_OK;
291 
292  // done func
293  if (dstream->func.writ.func((tb_astream_t*)dstream, state, dstream->func.writ.data, real, dstream->func.writ.size, dstream->func.writ.priv))
294  {
295  // not finished?
296  if (real < dstream->func.writ.size)
297  {
298  // update data and size
299  dstream->func.writ.data += real;
300  dstream->func.writ.size -= real;
301 
302  // continue to post writ
303  tb_aico_task_run(aice->aico, 0, tb_astream_data_writ_func, (tb_astream_t*)dstream);
304  }
305  }
306  }
307  break;
308  // killed
309  case TB_STATE_KILLED:
310  state = TB_STATE_KILLED;
311  break;
312  default:
313  tb_trace_d("read: unknown state: %s", tb_state_cstr(aice->state));
314  break;
315  }
316 
317  // done func
318  if (state != TB_STATE_OK) dstream->func.writ.func((tb_astream_t*)dstream, state, dstream->func.writ.data, 0, dstream->func.writ.size, dstream->func.writ.priv);
319 
320  // ok
321  return tb_true;
322 }

492 
493  // clear offset

495 
496  // check
497  tb_assert_and_check_return_val(dstream->data && dstream->size, tb_false);
498  return tb_true;
499  }
500  case TB_STREAM_CTRL_SET_URL:
501  {
502  // check
503  tb_assert_and_check_return_val(!tb_stream_is_opened(astream), tb_false);
504
