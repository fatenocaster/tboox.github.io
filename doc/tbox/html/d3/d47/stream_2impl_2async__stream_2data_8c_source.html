278 static tb_bool_t tb_async_stream_data_impl_read_func(tb_aice_t const* aice)
279 {
280  // check
281  tb_assert_and_check_return_val(aice && aice->aico && aice->code == TB_AICE_CODE_RUNTASK, tb_false);
282 
283  // the stream
284  tb_async_stream_data_impl_t* impl = (tb_async_stream_data_impl_t*)aice->priv;
285  tb_assert_and_check_return_val(impl && impl->func.read.func, tb_false);
286 
287  // done state
288  tb_size_t state = TB_STATE_UNKNOWN_ERROR;
289  switch (aice->state)
290  {
291  // ok
292  case TB_STATE_OK:
293  {
294  // check
295  tb_assert_and_check_break(impl->data && impl->head);
296 
297  // the left
298  tb_size_t left = impl->data + impl->size - impl->head;
299 
300  // the real
301  tb_size_t real = impl->func.read.size;
302  if (!real || real > left) real = left;
303 
304  // no data? closed
305  if (!real)
306  {
307  state = TB_STATE_CLOSED;
308  break;
309  }
310 
311  // save data
312  tb_byte_t* data = impl->head;
313 
314  // save head
315  impl->head += real;
316 
317  // save offset
318  tb_atomic64_set(&impl->offset, impl->head - impl->data);
319 
320  // ok
321  state = TB_STATE_OK;
322 
323  // done func
324  if (impl->func.read.func((tb_async_stream_ref_t)impl, state, data, real, impl->func.read.size, impl->func.read.priv))
325  {
326  // continue to post read
327  tb_aico_task_run(aice->aico, 0, tb_async_stream_data_impl_read_func, (tb_async_stream_ref_t)impl);
328  }
329  }
330  break;
331  // killed
332  case TB_STATE_KILLED:
333  state = TB_STATE_KILLED;
334  break;
335  default:
336  tb_trace_d("read: unknown state: %s", tb_state_cstr(aice->state));
337  break;
338  }
339 
340  // done func
341  if (state != TB_STATE_OK) impl->func.read.func((tb_async_stream_ref_t)impl, state, tb_null, 0, impl->func.read.size, impl->func.read.priv);
342 
343  // ok
344  return tb_true;
345 }

365  // the stream

367  tb_assert_and_check_return_val(impl && impl->func.writ.func, tb_false);
368 
369  // done state
370  tb_size_t state = TB_STATE_UNKNOWN_ERROR;
371  switch (aice->state)
372  {
373  // ok
374  case TB_STATE_OK:
375  {
376  // check
377  tb_assert_and_check_break(impl->data && impl->head);
378  tb_assert_and_check_break(impl->func.writ.data && impl->func.writ.size);
379 
380  // the left
381  tb_size_t left = impl->data + impl->size - impl->head;
382 
383  // the real
384  tb_size_t real = impl->func.writ.size;
385  if (real > left) real = left;

389  {
390  state = TB_STATE_CLOSED;
391  break;
392  }
393 
394  // save data
395  tb_memcpy(impl->head, impl->func.writ.data, real);
396 
397  // save head
398  impl->head += real;
399 
400  // save offset
401  tb_atomic64_set(&impl->offset, impl->head - impl->data);
402 
403  // ok
404  state = TB_STATE_OK;
405 
406  // done func
407  if (impl->func.writ.func((tb_async_stream_ref_t)impl, state, impl->func.writ.data, real, impl->func.writ.size, impl->func.writ.priv))
408  {
409  // not finished?
410  if (real < impl->func.writ.size)
411  {
412  // update data and size
413  impl->func.writ.data += real;
414  impl->func.writ.size -= real;
415 
416  // continue to post writ

418  }
419  }
420  }
421  break;

426  default:

429  }
430
