159 static tb_long_t tb_gstream_filter_read(tb_handle_t gstream, tb_byte_t* data, tb_size_t size)
160 {
161  // check
162  tb_gstream_filter_t* fstream = tb_gstream_filter_cast(gstream);
163  tb_assert_and_check_return_val(fstream && fstream->gstream, -1);
164 
165  // read
166  tb_long_t real = tb_gstream_aread(fstream->gstream, data, size);
167 
168  // done filter
169  if (fstream->filter)
170  {
171  // save mode: read
172  if (!fstream->mode) fstream->mode = 1;
173 
174  // check mode
175  tb_assert_and_check_return_val(fstream->mode == 1, -1);
176 
177  // save last
178  fstream->last = real;
179 
180  // eof?
181  if (real < 0 || (!real && fstream->wait) || tb_filter_beof(fstream->filter))
182  fstream->beof = tb_true;
183  // clear wait
184  else if (real > 0) fstream->wait = tb_false;
185 
186  // spak data
187  tb_byte_t const* odata = tb_null;
188  if (real) real = tb_filter_spak(fstream->filter, data, real < 0? 0 : real, &odata, size, fstream->beof? -1 : 0);
189  // no data? try to sync it
190  if (!real) real = tb_filter_spak(fstream->filter, tb_null, 0, &odata, size, fstream->beof? -1 : 1);
191 
192  // has data? save it
193  if (real > 0 && odata) tb_memcpy(data, odata, real);
194 
195  // eof?
196  if (fstream->beof && !real) real = -1;
197  }
198 
199  // ok?
200  return real;
201 }

260 static tb_long_t tb_gstream_filter_wait(tb_handle_t gstream, tb_size_t wait, tb_long_t timeout)
261 {
262  // check
263  tb_gstream_filter_t* fstream = tb_gstream_filter_cast(gstream);
264  tb_assert_and_check_return_val(fstream && fstream->gstream, -1);
265 
266  // done
267  tb_long_t ok = -1;
268  if (fstream->filter && fstream->mode == 1)
269  {
270  // wait ok
271  if (fstream->last > 0) ok = wait;
272  // need wait
273  else if (!fstream->last && !fstream->beof && !tb_filter_beof(fstream->filter))
274  {
275  // wait
276  ok = tb_gstream_wait(fstream->gstream, wait, timeout);
277 
278  // eof?
279  if (!ok)
280  {
281  // wait ok and continue to read or writ
282  ok = wait;
283 
284  // set eof
285  fstream->beof = tb_true;
286  }
287  // wait ok
288  else fstream->wait = tb_true;
289  }
290  // eof
291  else
292  {
293  // wait ok and continue to read or writ
294  ok = wait;
295 
296  // set eof
297  fstream->beof = tb_true;
298  }
299  }
300  else ok = tb_gstream_wait(fstream->gstream, wait, timeout);
301 
302  // ok?
303  return ok;
304 }
