154 tb_long_t tb_charset_conv_bst(tb_size_t ftype, tb_size_t ttype, tb_static_stream_ref_t fst, tb_static_stream_ref_t tst)
155 {
156  // check
157  tb_assert_and_check_return_val(TB_CHARSET_TYPE_OK(ftype) && TB_CHARSET_TYPE_OK(ttype) && fst && tst, -1);
158  tb_assert_and_check_return_val(tb_static_stream_valid(fst) && tb_static_stream_valid(tst), -1);
159 
160  // init the charset
161  tb_charset_ref_t fr = tb_charset_find_by_type(ftype);
162  tb_charset_ref_t to = tb_charset_find_by_type(ttype);
163  tb_assert_and_check_return_val(fr && to && fr->get && fr->set, -1);
164 
165  // no data?
166  tb_check_return_val(tb_static_stream_left(fst), 0);
167 
168  // big endian?
169  tb_bool_t fbe = !(ftype & TB_CHARSET_TYPE_LE)? tb_true : tb_false;
170  tb_bool_t tbe = !(ttype & TB_CHARSET_TYPE_LE)? tb_true : tb_false;
171 
172  // walk
173  tb_uint32_t ch;
174  tb_byte_t const* tp = tb_static_stream_pos(tst);
175  while (tb_static_stream_left(fst) && tb_static_stream_left(tst))
176  {
177  // get ucs4 character
178  tb_long_t ok = 0;
179  if ((ok = fr->get(fst, fbe, &ch)) > 0)
180  {
181  // set ucs4 character
182  if (to->set(tst, tbe, ch) < 0) break;
183  }
184  else if (ok < 0) break;
185  }
186 
187  // ok?
188  return tb_static_stream_pos(tst) - tp;
189 }
