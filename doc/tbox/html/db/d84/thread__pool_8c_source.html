381 static tb_pointer_t tb_thread_pool_worker_loop(tb_cpointer_t priv)
382 {
383  // the worker
384  tb_thread_pool_worker_t* worker = (tb_thread_pool_worker_t*)priv;
385 
386  // trace
387  tb_trace_d("worker[%lu]: init", worker? worker->id : -1);
388 
389  // done
390  do
391  {
392  // check
393  tb_assert_and_check_break(worker && !worker->jobs && !worker->stats);
394 
395  // the pool
396  tb_thread_pool_impl_t* impl = (tb_thread_pool_impl_t*)worker->pool;
397  tb_assert_and_check_break(impl && impl->semaphore);
398 
399  // wait some time for leaving the lock
400  tb_msleep((worker->id + 1)* 20);
401 
402  // init jobs
403  worker->jobs = tb_vector_init(TB_THREAD_POOL_JOBS_WORKING_GROW, tb_item_func_ptr(tb_null, tb_null));
404  tb_assert_and_check_break(worker->jobs);
405 
406  // init stats
407  worker->stats = tb_hash_init(TB_HASH_BULK_SIZE_MICRO, tb_item_func_ptr(tb_null, tb_null), tb_item_func_mem(sizeof(tb_thread_pool_job_stats_t), tb_null, tb_null));
408  tb_assert_and_check_break(worker->stats);
409 
410  // loop
411  while (1)
412  {
413  // pull jobs if be idle
414  if (!tb_vector_size(worker->jobs))
415  {
416  // enter
417  tb_spinlock_enter(&impl->lock);
418 
419  // init the pull time
420  worker->pull = 0;
421 
422  // pull from the urgent jobs
423  if (tb_list_entry_size(&impl->jobs_urgent))
424  {
425  // trace
426  tb_trace_d("worker[%lu]: try pulling from urgent: %lu", worker->id, tb_list_entry_size(&impl->jobs_urgent));
427 
428  // pull it
429  tb_remove_if(tb_list_entry_itor(&impl->jobs_urgent), tb_thread_pool_worker_walk_pull, worker);
430  }
431 
432  // pull from the waiting jobs
433  if (tb_list_entry_size(&impl->jobs_waiting))
434  {
435  // trace
436  tb_trace_d("worker[%lu]: try pulling from waiting: %lu", worker->id, tb_list_entry_size(&impl->jobs_waiting));
437 
438  // pull it
439  tb_remove_if(tb_list_entry_itor(&impl->jobs_waiting), tb_thread_pool_worker_walk_pull, worker);
440  }
441 
442  // pull from the pending jobs and clean some finished and killed jobs
443  if (tb_list_entry_size(&impl->jobs_pending))
444  {
445  // trace
446  tb_trace_d("worker[%lu]: try pulling from pending: %lu", worker->id, tb_list_entry_size(&impl->jobs_pending));
447 
448  // no jobs? try to pull from the pending jobs
449  if (!tb_vector_size(worker->jobs))
450  tb_remove_if(tb_list_entry_itor(&impl->jobs_pending), tb_thread_pool_worker_walk_pull_and_clean, worker);
451  // clean some finished and killed jobs
452  else tb_remove_if(tb_list_entry_itor(&impl->jobs_pending), tb_thread_pool_worker_walk_clean, worker);
453  }
454 
455  // leave
456  tb_spinlock_leave(&impl->lock);
457 
458  // idle? wait it
459  if (!tb_vector_size(worker->jobs))
460  {
461  // killed?
462  tb_check_break(!tb_atomic_get(&worker->bstoped));
463 
464  // trace
465  tb_trace_d("worker[%lu]: wait: ..", worker->id);
466 
467  // wait some time
468  tb_long_t wait = tb_semaphore_wait(impl->semaphore, -1);
469  tb_assert_and_check_break(wait > 0);
470 
471  // trace
472  tb_trace_d("worker[%lu]: wait: ok", worker->id);
473 
474  // continue it
475  continue;
476  }
477  else
478  {
479 #ifdef TB_TRACE_DEBUG
480  // update the jobs urgent size
481  tb_size_t jobs_urgent_size = tb_list_entry_size(&impl->jobs_urgent);
482 
483  // update the jobs waiting size
484  tb_size_t jobs_waiting_size = tb_list_entry_size(&impl->jobs_waiting);
485 
486  // update the jobs pending size
487  tb_size_t jobs_pending_size = tb_list_entry_size(&impl->jobs_pending);
488 
489  // trace
490  tb_trace_d("worker[%lu]: pull: jobs: %lu, time: %lu ms, waiting: %lu, pending: %lu, urgent: %lu", worker->id, tb_vector_size(worker->jobs), worker->pull, jobs_waiting_size, jobs_pending_size, jobs_urgent_size);
491 #endif
492  }
493  }
494 
495  // done jobs
496  tb_for_all (tb_thread_pool_job_t*, job, worker->jobs)
497  {
498  // check
499  tb_assert_and_check_continue(job && job->task.done);
500 
501  // the job state
502  tb_size_t state = tb_atomic_fetch_and_pset(&job->state, TB_STATE_WAITING, TB_STATE_WORKING);
503 
504  // the job is waiting? work it
505  if (state == TB_STATE_WAITING)
506  {
507  // trace
508  tb_trace_d("worker[%lu]: done: task[%p:%s]: ..", worker->id, job->task.done, job->task.name);
509 
510  // init the time
511  tb_hong_t time = tb_cache_time_spak();
512 
513  // done the job
514  job->task.done((tb_thread_pool_worker_ref_t)worker, job->task.priv);
515 
516  // computate the time
517  time = tb_cache_time_spak() - time;
518 
519  // exists? update time and count
520  tb_size_t itor;
521  tb_hash_item_t const* item = tb_null;
522  if ( ((itor = tb_hash_itor(worker->stats, job->task.done)) != tb_iterator_tail(worker->stats))
523  && (item = (tb_hash_item_t const*)tb_iterator_item(worker->stats, itor)))
524  {
525  // the stats
526  tb_thread_pool_job_stats_t* stats = (tb_thread_pool_job_stats_t*)item->data;
527  tb_assert_and_check_break(stats);
528 
529  // update the done count
530  stats->done_count++;
531 
532  // update the total time
533  stats->total_time += time;
534  }
535 
536  // no item? add it
537  if (!item)
538  {
539  // init stats
540  tb_thread_pool_job_stats_t stats = {0};
541  stats.done_count = 1;
542  stats.total_time = time;
543 
544  // add stats
545  tb_hash_set(worker->stats, job->task.done, &stats);
546  }
547 
548 #ifdef TB_TRACE_DEBUG
549  tb_size_t done_count = 0;
550  tb_hize_t total_time = 0;
551  tb_thread_pool_job_stats_t* stats = (tb_thread_pool_job_stats_t*)tb_hash_get(worker->stats, job->task.done);
552  if (stats)
553  {
554  done_count = stats->done_count;
555  total_time = stats->total_time;
556  }
557 
558  // trace
559  tb_trace_d("worker[%lu]: done: task[%p:%s]: time: %lld ms, average: %lld ms, count: %lu", worker->id, job->task.done, job->task.name, time, (total_time / (tb_hize_t)done_count), done_count);
560 #endif
561 
562  // update the job state
563  tb_atomic_set(&job->state, TB_STATE_FINISHED);
564  }
565  // the job is killing? work it
566  else if (state == TB_STATE_KILLING)
567  {
568  // update the job state
569  tb_atomic_set(&job->state, TB_STATE_KILLED);
570  }
571  }
572 
573  // clear jobs
574  tb_vector_clear(worker->jobs);
575  }
576 
577  } while (0);
578 
579  // exit worker
580  if (worker)
581  {
582  // trace
583  tb_trace_d("worker[%lu]: exit", worker->id);
584 
585  // stoped
586  tb_atomic_set(&worker->bstoped, 1);
587 
588  // exit all private data
589  tb_size_t i = 0;
590  tb_size_t n = tb_arrayn(worker->priv);
591  for (i = 0; i < n; i++)
592  {
593  // the private data
594  tb_thread_pool_worker_priv_t* priv = &worker->priv[n - i - 1];
595 
596  // exit it
597  if (priv->exit) priv->exit((tb_thread_pool_worker_ref_t)worker, priv->priv);
598 
599  // clear it
600  priv->exit = tb_null;
601  priv->priv = tb_null;
602  }
603 
604  // exit stats
605  if (worker->stats) tb_hash_exit(worker->stats);
606  worker->stats = tb_null;
607 
608  // exit jobs
609  if (worker->jobs) tb_vector_exit(worker->jobs);
610  worker->jobs = tb_null;
611  }
612 
613  // exit
614  tb_thread_return(tb_null);
615  return tb_null;
616 }
