138 {
139  tb_char_t const* p = ps;
140  tb_char_t* q = *pd;
141  while (p < se)
142  {
143  tb_byte_t c = *p++;
144  if (!c) break;
145  // is pointer? 11xxxxxx xxxxxxxx
146  else if (c >= 0xc0)
147  {
148  tb_uint16_t pos = c;
149  pos &= ~0xc0;
150  pos <<= 8;
151  pos |= *p++;
152  tb_dns_decode_name_impl(sb, se, sb + pos, &q);
153  break;
154  }
155  // is ascii? 00xxxxxx
156  else
157  {
158  while (c--) *q++ = *p++;
159  *q++ = '.';
160  }
161  }
162  *pd = q;
