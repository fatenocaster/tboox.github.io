80 
81  // the aiop
82  tb_aiop_t* aiop = reactor->aiop;
83  tb_assert_and_check_return_val(aiop, tb_false);
84 
85  // check size

87  tb_size_t size = tb_hash_size(rtor->hash);
88  if (rtor->mutx.hash) tb_mutex_leave(rtor->mutx.hash);

91  // add handle => aioo

93  if (rtor->mutx.hash) tb_mutex_enter(rtor->mutx.hash);

97  ok = tb_true;
98  }
99  if (rtor->mutx.hash) tb_mutex_leave(rtor->mutx.hash);
100  tb_assert_and_check_return_val(ok, tb_false);
101 
102  // the fd
103  tb_long_t fd = ((tb_long_t)aioo->handle) - 1;
104 
105  // the code
106  tb_size_t code = aioo->code;
107 
108  // enter
109  if (rtor->mutx.pfds) tb_mutex_enter(rtor->mutx.pfds);
110 
111  // update fd max
112  if (fd > rtor->sfdm) rtor->sfdm = fd;
113 
114  // init fds
115  if (code & (TB_AIOE_CODE_RECV | TB_AIOE_CODE_ACPT)) FD_SET(fd, &rtor->rfdi);
116  if (code & (TB_AIOE_CODE_SEND | TB_AIOE_CODE_CONN)) FD_SET(fd, &rtor->wfdi);
117  FD_SET(fd, &rtor->efdi);
118 
119  // leave
120  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
121 
122  // spak it
123  if (aiop->spak[0] && code) tb_socket_send(aiop->spak[0], "p", 1);
124 
125  // ok?
126  return ok;
127 }

193  FD_SET(fd, &rtor->efdi);
194  }
195  else
196  {
197  FD_CLR(fd, &rtor->efdi);
198  }
199 
200  // leave
201  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
202 
203  // spak it

209 static tb_long_t tb_aiop_reactor_select_wait(tb_aiop_reactor_t* reactor, tb_aioe_t* list, tb_size_t maxn, tb_long_t timeout)
210 {
211  // check
212  tb_aiop_reactor_select_t* rtor = (tb_aiop_reactor_select_t*)reactor;
213  tb_assert_and_check_return_val(rtor && reactor->aiop && list && maxn, -1);
214 
215  // the aiop
216  tb_aiop_t* aiop = reactor->aiop;
217  tb_assert_and_check_return_val(aiop, tb_false);
218 
219  // init time
220  struct timeval t = {0};
221  if (timeout > 0)
222  {
223  t.tv_sec = timeout / 1000;
224  t.tv_usec = (timeout % 1000) * 1000;
225  }
226 
227  // loop
228  tb_long_t wait = 0;
229  tb_bool_t stop = tb_false;
230  tb_hong_t time = tb_mclock();
231  while (!wait && !stop && (timeout < 0 || tb_mclock() < time + timeout))
232  {
233  // enter
234  if (rtor->mutx.pfds) tb_mutex_enter(rtor->mutx.pfds);
235 
236  // init fdo
237  tb_size_t sfdm = rtor->sfdm;
238  tb_memcpy(&rtor->rfdo, &rtor->rfdi, sizeof(fd_set));
239  tb_memcpy(&rtor->wfdo, &rtor->wfdi, sizeof(fd_set));
240  tb_memcpy(&rtor->efdo, &rtor->efdi, sizeof(fd_set));
241 
242  // leave
243  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
244 
245  // wait
246  tb_long_t sfdn = select(sfdm + 1, &rtor->rfdo, &rtor->wfdo, &rtor->efdo, timeout >= 0? &t : tb_null);
247  tb_assert_and_check_return_val(sfdn >= 0, -1);
248 
249  // timeout?
250  tb_check_return_val(sfdn, 0);
251 
252  // enter
253  if (rtor->mutx.hash) tb_mutex_enter(rtor->mutx.hash);
254 
255  // sync
256  tb_size_t itor = tb_iterator_head(rtor->hash);
257  tb_size_t tail = tb_iterator_tail(rtor->hash);
258  for (; itor != tail && wait >= 0 && wait < maxn; itor = tb_iterator_next(rtor->hash, itor))
259  {
260  tb_hash_item_t* item = tb_iterator_item(rtor->hash, itor);
261  if (item)
262  {
263  // the handle
264  tb_handle_t handle = (tb_handle_t)item->name;
265  tb_assert_and_check_return_val(handle, -1);
266 
267  // spak?
268  if (handle == aiop->spak[1] && FD_ISSET(((tb_long_t)aiop->spak[1] - 1), &rtor->rfdo))
269  {
270  // read spak
271  tb_char_t spak = '\0';
272  if (1 != tb_socket_recv(aiop->spak[1], &spak, 1)) wait = -1;
273 
274  // killed?
275  if (spak == 'k') wait = -1;
276  tb_check_break(wait >= 0);
277 
278  // stop to wait
279  stop = tb_true;
280 
281  // continue it
282  continue ;
283  }
284 
285  // filter spak
286  tb_check_continue(handle != aiop->spak[1]);
287 
288  // the fd
289  tb_long_t fd = (tb_long_t)item->name - 1;
290 
291  // the aioo
292  tb_aioo_t* aioo = (tb_aioo_t*)item->data;
293  tb_assert_and_check_return_val(aioo && aioo->handle == handle, -1);
294 
295  // init aioe
296  tb_aioe_t aioe = {0};
297  aioe.data = aioo->data;
298  aioe.aioo = aioo;
299  if (FD_ISSET(fd, &rtor->rfdo))
300  {
301  aioe.code |= TB_AIOE_CODE_RECV;
302  if (aioo->code & TB_AIOE_CODE_ACPT) aioe.code |= TB_AIOE_CODE_ACPT;
303  }
304  if (FD_ISSET(fd, &rtor->wfdo))
305  {
306  aioe.code |= TB_AIOE_CODE_SEND;
307  if (aioo->code & TB_AIOE_CODE_CONN) aioe.code |= TB_AIOE_CODE_CONN;
308  }
309  if (FD_ISSET(fd, &rtor->efdo) && !(aioe.code & (TB_AIOE_CODE_RECV | TB_AIOE_CODE_SEND)))
310  aioe.code |= TB_AIOE_CODE_RECV | TB_AIOE_CODE_SEND;
311 
312  // ok?
313  if (aioe.code)
314  {
315  // save aioe
316  list[wait++] = aioe;
317 
318  // oneshot? clear it
319  if (aioo->code & TB_AIOE_CODE_ONESHOT)
320  {
321  // clear aioo
322  aioo->code = TB_AIOE_CODE_NONE;
323  aioo->data = tb_null;
324 
325  // clear events
326  if (rtor->mutx.pfds) tb_mutex_enter(rtor->mutx.pfds);
327  FD_CLR(fd, &rtor->rfdi);
328  FD_CLR(fd, &rtor->wfdi);
329  FD_CLR(fd, &rtor->efdi);
330  if (rtor->mutx.pfds) tb_mutex_leave(rtor->mutx.pfds);
331  }
332  }
333  }
334  }
335 
336  // leave
337  if (rtor->mutx.hash) tb_mutex_leave(rtor->mutx.hash);
338  }
339 
340  // ok
341  return wait;
342 }
