694  tb_size_t push = 0;
695  tb_size_t need = size - tb_queue_buffer_size(&impl->cache);
696  tb_byte_t* tail = tb_queue_buffer_push_init(&impl->cache, &push);
697  tb_assert_and_check_return_val(tail && push, tb_false);
698  if (push > need) push = need;
699

701  tb_size_t read = 0;
702  while (read < push && (TB_STATE_OPENED == tb_atomic_get(&impl->istate)))
703  {
704  // read data
705  tb_long_t real = impl->read(stream, tail + read, push - read);
706 
707  // ok?
708  if (real > 0)
709  {
710  // save read
711  read += real;
712  }
713  // no data?
714  else if (!real)
715  {
716  // wait
717  real = impl->wait(stream, TB_STREAM_WAIT_READ, tb_stream_timeout(stream));
718 
719  // ok?
720  tb_check_break(real > 0);
721  }
722  else break;
723  }
724 
725  // leave cache for push
726  tb_queue_buffer_push_exit(&impl->cache, read);
727 
728  // not enough?
729  if (size > tb_queue_buffer_size(&impl->cache))
730  {
731  // killed? save state
732  if (!impl->state && (TB_STATE_KILLING == tb_atomic_get(&impl->istate)))
733  impl->state = TB_STATE_KILLED;
734 
735  // failed
736  return tb_false;
737  }
738 
739  // save data
740  *data = tb_queue_buffer_head(&impl->cache);
741 
742  // ok
743  return tb_true;
744 }
745 tb_long_t tb_stream_read(tb_stream_ref_t stream, tb_byte_t* data, tb_size_t size)
746 {

772

787  tb_check_return_val(real >= 0, -1);
788

791  {
792  // leave cache for push
793  tb_queue_buffer_push_exit(&impl->cache, real);
794 
795  // read cache
796  real = tb_queue_buffer_read(&impl->cache, data + read, tb_min(real, size - read));
797  tb_check_return_val(real >= 0, -1);
798 
799  // save read

801  }
802  }
803  else
804  {
805  // read it directly

807  tb_check_return_val(read >= 0, -1);
808  }
809  }
810  while (0);
811

813  impl->offset += read;

816  return read;

1033 tb_bool_t tb_stream_seek(tb_stream_ref_t stream, tb_hize_t offset)
1034 {
1035  // check
1036  tb_stream_impl_t* impl = tb_stream_impl(stream);
1037  tb_assert_and_check_return_val(impl && tb_stream_is_opened(stream), tb_false);
1038 
1039  // stoped?
1040  tb_assert_and_check_return_val((TB_STATE_OPENED == tb_atomic_get(&impl->istate)), tb_false);
1041 
1042  // limit offset
1043  tb_hong_t size = tb_stream_size(stream);
1044  if (size >= 0 && offset > size) offset = size;
1045 
1046  // the offset be not changed?
1047  tb_hize_t curt = tb_stream_offset(stream);
1048  tb_check_return_val(offset != curt, tb_true);
1049 
1050  // for writing
1051  if (impl->bwrited)
1052  {
1053  // sync writed data first
1054  if (!tb_stream_sync(stream, tb_false)) return tb_false;
1055 
1056  // check cache, must not cache or empty cache
1057  tb_assert_and_check_return_val(!tb_queue_buffer_maxn(&impl->cache) || tb_queue_buffer_null(&impl->cache), tb_false);
1058 
1059  // seek it
1060  tb_bool_t ok = impl->seek? impl->seek(stream, offset) : tb_false;
1061 
1062  // save offset
1063  if (ok) impl->offset = offset;
1064  }
1065  // for reading
1066  else
1067  {
1068  // cached? try to seek it at the cache
1069  tb_bool_t ok = tb_false;
1070  if (tb_queue_buffer_maxn(&impl->cache))
1071  {
1072  tb_size_t size = 0;
1073  tb_byte_t* data = tb_queue_buffer_pull_init(&impl->cache, &size);
1074  if (data && size && offset > curt && offset < curt + size)
1075  {
1076  // seek it at the cache
1077  tb_queue_buffer_pull_exit(&impl->cache, (tb_size_t)(offset - curt));
1078 
1079  // save offset
1080  impl->offset = offset;
1081 
1082  // ok
1083  ok = tb_true;
1084  }
1085  }
1086 
1087  // seek it
1088  if (!ok)
1089  {
1090  // seek it
1091  ok = impl->seek? impl->seek(stream, offset) : tb_false;
1092 
1093  // ok?
1094  if (ok)
1095  {
1096  // save offset
1097  impl->offset = offset;
1098 
1099  // clear cache
1100  tb_queue_buffer_clear(&impl->cache);
1101  }
1102  }
1103 
1104  // try to read and seek
1105  if (!ok && offset > curt)
1106  {
1107  // read some data for updating offset
1108  tb_byte_t data[TB_STREAM_BLOCK_MAXN];
1109  while (tb_stream_offset(stream) != offset)
1110  {
1111  tb_size_t need = (tb_size_t)tb_min(offset - curt, TB_STREAM_BLOCK_MAXN);
1112  if (!tb_stream_bread(stream, data, need)) return tb_false;
1113  }
1114  }
1115  }
1116 
1117  // ok?
1118  return tb_stream_offset(stream) == offset? tb_true : tb_false;
1119 }
