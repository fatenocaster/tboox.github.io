456

462  }

464  case TB_XML_READER_EVENT_ELEMENT_BEG:

468  tb_assert_and_check_goto(name, end);

471  tb_sstring_chrcat(&s, '/');
472  tb_sstring_cstrcat(&s, name);
473 
474  // ok?

476  tb_trace_impl("path: %s", tb_sstring_cstr(&s));
477

497  break;
498  }

502  }
503 
504 end:
