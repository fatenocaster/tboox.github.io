107  t = SOCK_STREAM;

109  }

115  else break;
116

121  // save sock

123 
124  // non-block

126  if (tb_ws2_32()->ioctlsocket(fd, FIONBIO, &nb) == SOCKET_ERROR) break;

130 
131  } while (0);
132 
133  // failed?
134  if (!ok)
135  {
136  // exit it

139  }

144 tb_bool_t tb_socket_pair(tb_size_t type, tb_socket_ref_t pair[2])
145 {
146  // check
147  tb_assert_and_check_return_val(type && pair, tb_false);
148 
149  // init pair
150  pair[0] = tb_null;
151  pair[1] = tb_null;
152 
153  // init type & protocol
154  tb_int_t t = 0;
155  tb_int_t p = 0;
156  switch (type)
157  {
158  case TB_SOCKET_TYPE_TCP:
159  {
160  t = SOCK_STREAM;
161  p = IPPROTO_TCP;
162  }
163  break;
164  case TB_SOCKET_TYPE_UDP:
165  {
166  t = SOCK_DGRAM;
167  p = IPPROTO_UDP;
168  }
169  break;
170  default:
171  return tb_false;
172  }
173 
174  // done
175  tb_bool_t ok = tb_false;
176  SOCKET listener = INVALID_SOCKET;
177  SOCKET sock1 = INVALID_SOCKET;
178  SOCKET sock2 = INVALID_SOCKET;
179  do
180  {
181  // init listener
182  listener = tb_ws2_32()->WSASocketA(AF_INET, t, p, tb_null, 0, WSA_FLAG_OVERLAPPED);
183  tb_assert_and_check_break(listener != INVALID_SOCKET);
184 
185  // init bind address
186  SOCKADDR_IN b = {0};
187  b.sin_family = AF_INET;
188  b.sin_port = 0;
189  b.sin_addr.S_un.S_addr = tb_bits_ne_to_be_u32(INADDR_LOOPBACK);
190 
191  // reuse addr
192 #ifdef SO_REUSEADDR
193  {
194  tb_int_t reuseaddr = 1;
195  if (tb_ws2_32()->setsockopt(listener, SOL_SOCKET, SO_REUSEADDR, (tb_char_t*)&reuseaddr, sizeof(reuseaddr)) < 0)
196  break;
197  }
198 #endif
199 
200  // bind it
201  if (tb_ws2_32()->bind(listener, (struct sockaddr *)&b, sizeof(b)) == SOCKET_ERROR) break;
202 
203  // get socket address
204  SOCKADDR_IN d = {0};
205  tb_int_t n = sizeof(SOCKADDR_IN);
206  if (tb_ws2_32()->getsockname(listener, (struct sockaddr *)&d, &n) == SOCKET_ERROR) break;
207  d.sin_addr.S_un.S_addr = tb_bits_ne_to_be_u32(INADDR_LOOPBACK);
208  d.sin_family = AF_INET;
209 
210  // listen it
211  if (tb_ws2_32()->listen(listener, 1) == SOCKET_ERROR) break;
212 
213  // init sock1
214  sock1 = tb_ws2_32()->WSASocketA(AF_INET, t, p, tb_null, 0, WSA_FLAG_OVERLAPPED);
215  tb_assert_and_check_break(sock1 != INVALID_SOCKET);
216 
217  // connect it
218  if (tb_ws2_32()->connect(sock1, (struct sockaddr const*)&d, sizeof(d)) == SOCKET_ERROR) break;
219 
220  // accept it
221  sock2 = tb_ws2_32()->accept(listener, tb_null, tb_null);
222  tb_assert_and_check_break(sock2 != INVALID_SOCKET);
223 
224  // set non-block
225  tb_ulong_t nb = 1;
226  if (tb_ws2_32()->ioctlsocket(sock1, FIONBIO, &nb) == SOCKET_ERROR) break;
227  if (tb_ws2_32()->ioctlsocket(sock2, FIONBIO, &nb) == SOCKET_ERROR) break;
228 
229  // ok
230  ok = tb_true;
231 
232  } while (0);
233 
234  // exit listener
235  if (listener != INVALID_SOCKET) tb_ws2_32()->closesocket(listener);
236  listener = INVALID_SOCKET;
237 
238  // failed? exit it
239  if (!ok)
240  {
241  // exit sock1
242  if (sock1 != INVALID_SOCKET) tb_ws2_32()->closesocket(sock1);
243  sock1 = INVALID_SOCKET;
244 
245  // exit sock2
246  if (sock2 != INVALID_SOCKET) tb_ws2_32()->closesocket(sock2);
247  sock2 = INVALID_SOCKET;
248  }
249  else
250  {
251  pair[0] = (tb_socket_ref_t)(sock1 + 1);
252  pair[1] = (tb_socket_ref_t)(sock2 + 1);
253  }
254 
255  // ok?
256  return ok;
257 }
