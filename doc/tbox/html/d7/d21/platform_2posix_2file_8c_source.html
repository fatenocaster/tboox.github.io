46 tb_file_ref_t tb_file_init(tb_char_t const* path, tb_size_t mode)
47 {

50 
51  // the full path
52  tb_char_t full[TB_PATH_MAXN];
53  path = tb_path_full(path, full, TB_PATH_MAXN);
54  tb_assert_and_check_return_val(path, tb_null);
55 
56  // flags
57  tb_size_t flags = 0;
58  if (mode & TB_FILE_MODE_RO) flags |= O_RDONLY;
59  else if (mode & TB_FILE_MODE_WO) flags |= O_WRONLY;
60  else if (mode & TB_FILE_MODE_RW) flags |= O_RDWR;
61 
62  if (mode & TB_FILE_MODE_CREAT) flags |= O_CREAT;
63  if (mode & TB_FILE_MODE_APPEND) flags |= O_APPEND;
64  if (mode & TB_FILE_MODE_TRUNC) flags |= O_TRUNC;
65 
66  // dma mode, no cache
67 #ifdef TB_CONFIG_OS_LINUX
68  if (mode & TB_FILE_MODE_DIRECT) flags |= O_DIRECT;
69 #endif
70 
71  // for native aio aicp
72 #if defined(TB_CONFIG_ASIO_HAVE_NAIO)
73  if (mode & TB_FILE_MODE_ASIO) flags |= O_DIRECT;
74 #endif
75 
76  // noblock
77  flags |= O_NONBLOCK;
78 
79  // modes
80  tb_size_t modes = 0;
81  if (mode & TB_FILE_MODE_CREAT)
82  {
83  //if ((mode & TB_FILE_MODE_RO) | (mode & TB_FILE_MODE_RW)) modes |= S_IREAD;
84  //if ((mode & TB_FILE_MODE_WO) | (mode & TB_FILE_MODE_RW)) modes |= S_IWRITE;
85  modes = 0777;
86  }
87 
88  // open it
89  tb_long_t fd = open(path, flags, modes);
90  if (fd < 0 && (mode & TB_FILE_MODE_CREAT))
91  {
92  // make directory
93  tb_char_t temp[TB_PATH_MAXN] = {0};
94  tb_char_t const* p = full;
95  tb_char_t* t = temp;
96  tb_char_t const* e = temp + TB_PATH_MAXN - 1;
97  for (; t < e && *p; t++)
98  {
99  *t = *p;
100  if (*p == '/')
101  {
102  // make directory if not exists
103  if (!tb_file_info(temp, tb_null)) mkdir(temp, S_IRWXU);
104 
105  // skip repeat '/'
106  while (*p && *p == '/') p++;
107  }
108  else p++;
109  }
110 
111  // open it again
112  fd = open(path, flags, modes);
113  }
114 
115  // trace
116  tb_trace_d("open: %p", tb_fd2file(fd));
117 
118  // ok?
119  return tb_fd2file(fd);
120 }

208 tb_hong_t tb_file_writf(tb_file_ref_t file, tb_file_ref_t ifile, tb_hize_t offset, tb_hize_t size)
209 {
210  // check
211  tb_assert_and_check_return_val(file && ifile && size, -1);
212 
213 #if defined(TB_CONFIG_OS_LINUX) || defined(TB_CONFIG_OS_ANDROID)
214 
215  // writ it
216  off_t seek = offset;
217  tb_hong_t real = sendfile(tb_file2fd(file), tb_file2fd(ifile), &seek, (size_t)size);
218 
219  // ok?
220  if (real >= 0) return real;
221 
222  // continue?
223  if (errno == EINTR || errno == EAGAIN) return 0;
224 
225  // error
226  return -1;
227 
228 #else
229 
230  // read data
231  tb_byte_t data[8192];
232  tb_long_t read = tb_file_pread(ifile, data, sizeof(data), offset);
233  tb_check_return_val(read > 0, read);
234 
235  // writ data
236  tb_size_t writ = 0;
237  while (writ < read)
238  {
239  tb_long_t real = tb_file_writ(file, data + writ, read - writ);
240  if (real > 0) writ += real;
241  else break;
242  }
243 
244  // ok?
245  return writ == read? writ : -1;
246 #endif
247 }
