712  if (size)
713  {
714  if (!tb_gstream_bwrit(gstream, (tb_byte_t*)data, size)) return -1;

719  while (*p)

721  if (!tb_gstream_bwrit(gstream, (tb_byte_t*)p, 1)) return -1;
722  p++;
723  }
724 
725  writ = p - data;
726  }
727

729 #ifdef TB_CONFIG_OS_WINDOWS
730  tb_char_t le[] = "\r\n";
731  tb_size_t ln = 2;

734  tb_size_t ln = 1;
735 #endif
736  if (!tb_gstream_bwrit(gstream, (tb_byte_t*)le, ln)) return -1;
