197 tb_void_t tb_md5_spak(tb_md5_t* md5, tb_byte_t const* data, tb_size_t size)
198 {
199  // check
200  tb_assert_and_check_return(md5 && data);
201 
202  // init
203  tb_uint32_t ip[16];
204  tb_size_t i = 0, ii = 0;
205 
206  // compute number of bytes mod 64
207  tb_int_t mdi = (tb_int_t)((md5->i[0] >> 3) & 0x3F);
208 
209  // update number of bits
210  if ((md5->i[0] + ((tb_uint32_t)size << 3)) < md5->i[0]) md5->i[1]++;
211 
212  md5->i[0] += ((tb_uint32_t)size << 3);
213  md5->i[1] += ((tb_uint32_t)size >> 29);
214 
215  while (size--)
216  {
217  // add new character to buffer, increment mdi
218  md5->ip[mdi++] = *data++;
219 
220  // transform if necessary
221  if (mdi == 0x40)
222  {
223  for (i = 0, ii = 0; i < 16; i++, ii += 4)
224  {
225  ip[i] = (((tb_uint32_t)md5->ip[ii + 3]) << 24)
226  | (((tb_uint32_t)md5->ip[ii + 2]) << 16)
227  | (((tb_uint32_t)md5->ip[ii + 1]) << 8)
228  | ((tb_uint32_t)md5->ip[ii]);
229  }
230 
231  tb_md5_transform(md5->sp, ip);
232  mdi = 0;
233  }
234  }
