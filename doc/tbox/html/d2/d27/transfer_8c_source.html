65  tb_size_t writ1s = 0;
66  do
67  {

70

72  tb_long_t real = tb_stream_read(istream, data, need);
73  if (real > 0)
74  {
75  // writ data
76  if (!tb_stream_bwrit(ostream, data, real)) break;
77 
78  // save writ
79  writ += real;
80 
81  // has func or limit rate?
82  if (func || lrate)
83  {
84  // the time
85  time = tb_cache_time_spak();
86 
87  // < 1s?
88  if (time < base1s + 1000)
89  {
90  // save writ1s
91  writ1s += real;
92 
93  // save current rate if < 1s from base
94  if (time < base + 1000) crate = writ1s;
95 
96  // compute the delay for limit rate
97  if (lrate) delay = writ1s >= lrate? (tb_size_t)(base1s + 1000 - time) : 0;
98  }
99  else

102  crate = writ1s;
103 
104  // update base1s
105  base1s = time;
106 
107  // reset writ1s
108  writ1s = 0;
109 
110  // reset delay
111  delay = 0;
112 
113  // done func
114  if (func) func(TB_STATE_OK, tb_stream_offset(istream), tb_stream_size(istream), writ, crate, priv);
115  }
116 
117  // wait some time for limit rate
118  if (delay) tb_msleep(delay);
119  }
120  }
121  else if (!real)
122  {
123  // wait

126

128  tb_check_break(wait);
129 
130  // has writ?
131  tb_assert_and_check_break(wait & TB_STREAM_WAIT_READ);
132  }
133  else break;
134 
135  // is end?
136  if (left && writ >= left) break;
137 
138  } while(1);
139 
140  // sync the ostream

142 
143  // has func?
144  if (func)
145  {
146  // the time
147  time = tb_cache_time_spak();
148 
149  // compute the total rate
150  tb_size_t trate = (writ && (time > base))? (tb_size_t)((writ * 1000) / (time - base)) : (tb_size_t)writ;
151 
152  // done func

154  }
155 
156  // ok?
