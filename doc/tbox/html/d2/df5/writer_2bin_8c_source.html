328 static tb_long_t tb_object_bin_writer_done(tb_stream_ref_t stream, tb_object_ref_t object, tb_bool_t deflate)
329 {
330  // check
331  tb_assert_and_check_return_val(object && stream, -1);
332 
333  // the func
334  tb_object_bin_writer_func_t func = tb_object_bin_writer_func(object->type);
335  tb_assert_and_check_return_val(func, -1);
336 
337  // the begin offset
338  tb_hize_t bof = tb_stream_offset(stream);
339 
340  // writ bin header
341  if (!tb_stream_bwrit(stream, (tb_byte_t const*)"tbo00", 5)) return -1;
342 
343  // done
344  tb_object_bin_writer_t writer = {0};
345  do
346  {
347  // init writer
348  writer.stream = stream;
349  writer.ohash = tb_hash_init(TB_HASH_BULK_SIZE_MICRO, tb_item_func_ptr(tb_null, tb_null), tb_item_func_uint32());
350  writer.shash = tb_hash_init(TB_HASH_BULK_SIZE_MICRO, tb_item_func_str(tb_true), tb_item_func_uint32());
351  tb_assert_and_check_break(writer.shash && writer.ohash);
352 
353  // writ
354  if (!func(&writer, object)) break;
355 
356  // sync
357  if (!tb_stream_sync(stream, tb_true)) break;
358 
359  } while (0);
360 
361  // exit the hash
362  if (writer.ohash) tb_hash_exit(writer.ohash);
363  if (writer.shash) tb_hash_exit(writer.shash);
364 
365  // exit the data
366  if (writer.data) tb_free(writer.data);
367 
368  // the end offset
369  tb_hize_t eof = tb_stream_offset(stream);
370 
371  // ok?
372  return eof >= bof? (tb_long_t)(eof - bof) : -1;
373 }
