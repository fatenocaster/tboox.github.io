182 static tb_bool_t tb_gstream_data_ctrl(tb_handle_t gstream, tb_size_t ctrl, tb_va_list_t args)
183 {
184  // check
185  tb_gstream_data_t* dstream = tb_gstream_data_cast(gstream);
186  tb_assert_and_check_return_val(dstream, tb_false);
187 
188  // ctrl
189  switch (ctrl)
190  {
191  case TB_STREAM_CTRL_GET_SIZE:
192  {
193  // the psize
194  tb_hong_t* psize = (tb_hong_t*)tb_va_arg(args, tb_hong_t*);
195  tb_assert_and_check_return_val(psize, tb_false);
196 
197  // get size
198  *psize = dstream->size;
199  return tb_true;
200  }
201  case TB_STREAM_CTRL_GET_OFFSET:
202  {
203  // the poffset
204  tb_hize_t* poffset = (tb_hize_t*)tb_va_arg(args, tb_hize_t*);
205  tb_assert_and_check_return_val(poffset, tb_false);
206 
207  // get offset
208  *poffset = dstream->base.offset;
209  return tb_true;
210  }
211  case TB_STREAM_CTRL_DATA_SET_DATA:
212  {
213  // exit data first if exists
214  if (dstream->data && !dstream->bref) tb_free(dstream->data);
215 
216  // save data
217  dstream->data = (tb_byte_t*)tb_va_arg(args, tb_byte_t*);
218  dstream->size = (tb_size_t)tb_va_arg(args, tb_size_t);
219  dstream->head = tb_null;
220  dstream->bref = tb_true;
221 
222  // check
223  tb_assert_and_check_return_val(dstream->data && dstream->size, tb_false);
224  return tb_true;
225  }
226  case TB_STREAM_CTRL_SET_URL:
227  {
228  // check
229  tb_assert_and_check_return_val(!tb_stream_is_opened(gstream), tb_false);
230 
231  // set url
232  tb_char_t const* url = (tb_char_t const*)tb_va_arg(args, tb_char_t const*);
233  tb_assert_and_check_return_val(url, tb_false);
234 
235  // the url size
236  tb_size_t url_size = tb_strlen(url);
237  tb_assert_and_check_return_val(url_size > 7, tb_false);
238 
239  // the base64 data and size
240  tb_char_t const* base64_data = url + 7;
241  tb_size_t base64_size = url_size - 7;
242 
243  // make data
244  tb_size_t maxn = base64_size;
245  tb_byte_t* data = tb_malloc(maxn);
246  tb_assert_and_check_return_val(data, tb_false);
247 
248  // decode base64 data

250  tb_assert_and_check_return_val(size, tb_false);
251 
252  // exit data first if exists
253  if (dstream->data && !dstream->bref) tb_free(dstream->data);
254 
255  // save data
256  dstream->data = data;
257  dstream->size = size;
258  dstream->bref = tb_false;
259  dstream->head = tb_null;
260 
261  // ok
262  return tb_true;
263  }
264  break;

267  }
